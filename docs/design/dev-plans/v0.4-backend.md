```go
// internal/server/spell/manager.go

package spell

type Manager struct {
    spellData    *dnd5e.SpellData
    stateManager *state.Manager
    dice         *dice.Service
}

// CastSpell 施放法术
func (m *Manager) CastSpell(params CastParams) (*CastResult, error) {
    result := &CastResult{}

    // 获取施法者
    caster, err := m.stateManager.GetCharacter(params.CasterID)
    if err != nil {
        return nil, err
    }

    // 获取法术数据
    spell, err := m.spellData.Get(params.SpellID)
    if err != nil {
        return nil, err
    }

    // 验证施法者知道这个法术
    if !m.knowsSpell(caster, params.SpellID) {
        return nil, errors.New("caster does not know this spell")
    }

    // 验证法术位
    if spell.Level > 0 && !m.hasSpellSlot(caster, spell.Level) {
        return nil, errors.New("no available spell slot")
    }

    // 检查专注
    if spell.RequiresConcentration && caster.Spellcasting.Concentration != nil {
        // 需要结束当前专注
        m.endConcentration(caster)
    }

    // 处理法术攻击
    if spell.AttackType == "attack" {
        result.AttackRoll = m.spellAttack(caster, params.TargetID)
    }

    // 处理豁免
    if spell.SavingThrow != "" {
        result.SavingThrow = m.savingThrow(params.TargetID, spell, caster)
    }

    // 应用法术效果
    result.Effects = m.applySpellEffect(spell, caster, params.TargetID, params.CastLevel)

    // 消耗法术位
    if spell.Level > 0 {
        m.consumeSpellSlot(caster, params.CastLevel)
    }

    // 设置专注
    if spell.RequiresConcentration {
        m.startConcentration(caster, params.SpellID, spell.Duration)
    }

    return result, nil
}

// SpellSlots 获取法术位信息
func (m *Manager) GetSpellSlots(characterID string) (map[int]*SpellSlotInfo, error) {
    caster, err := m.stateManager.GetCharacter(characterID)
    if err != nil {
        return nil, err
    }

    if caster.Spellcasting == nil {
        return nil, errors.New("character cannot cast spells")
    }

    // 获取职业法术位进度
    classSlots := m.getClassSpellSlots(caster.Class, caster.Level)

    result := make(map[int]*SpellSlotInfo)
    for level, max := range classSlots {
        remaining := caster.Spellcasting.SpellSlots[level]
        result[level] = &SpellSlotInfo{
            Level:     level,
            Max:       max,
            Remaining: remaining,
        }
    }

    return result, nil
}

// RestoreSpellSlots 恢复法术位（休息时）
func (m *Manager) RestoreSpellSlots(characterID string, shortRest bool) error {
    caster, err := m.stateManager.GetCharacter(characterID)
    if err != nil {
        return err
    }

    if caster.Spellcasting == nil {
        return nil
    }

    if shortRest {
        // 短休息：术士等特定职业可以恢复部分法术位
        // 暂不实现
    } else {
        // 长休息：恢复所有法术位
        classSlots := m.getClassSpellSlots(caster.Class, caster.Level)
        for level, max := range classSlots {
            caster.Spellcasting.SpellSlots[level] = max
        }
    }

    return nil
}
```

#### 2. 法术效果

```go
// internal/server/spell/effects.go

func (m *Manager) applySpellEffect(spell *dnd5e.Spell, caster *models.Character, targetID string, castLevel int) []*SpellEffect {
    effects := make([]*SpellEffect, 0)

    switch spell.ID {
    case "magic_missile":
        // 魔法飞弹：自动命中，3d4+3 力场伤害
        // 升环：每高于1环2级增加一个飞弹
        missileCount := 3 + (castLevel-1)/2
        for i := 0; i < missileCount; i++ {
            roller, _ := dice.Parse("1d4+1")
            result := roller.Roll(m.dice.Rand())
            effects = append(effects, &SpellEffect{
                Type:   "damage",
                Value:  result.Total,
                DamageType: "force",
            })
        }

    case "fireball":
        // 火球术：8d6 火焰伤害，DC 豁免减半
        roller, _ := dice.Parse("8d6")
        result := roller.Roll(m.dice.Rand())
        damage := result.Total
        // 升环：每高于3环1级增加 1d6
        for i := 0; i < castLevel-3; i++ {
            extra, _ := dice.Parse("1d6")
            extraResult := extra.Roll(m.dice.Rand())
            damage += extraResult.Total
        }
        effects = append(effects, &SpellEffect{
            Type:       "damage",
            Value:      damage,
            DamageType: "fire",
            SaveHalf:   true,
        })

    case "cure_wounds":
        // 治疗术：1d8 + 施法属性修正
        roller, _ := dice.Parse("1d8")
        result := roller.Roll(m.dice.Rand())
        healing := result.Total + dice.GetModifier(caster.AbilityScores[caster.Spellcasting.Ability])
        // 升环：每高于1环1级增加 1d8
        for i := 0; i < castLevel-1; i++ {
            extra, _ := dice.Parse("1d8")
            extraResult := extra.Roll(m.dice.Rand())
            healing += extraResult.Total
        }
        effects = append(effects, &SpellEffect{
            Type:  "healing",
            Value: healing,
        })

    case "shield":
        // 护盾术：+5 AC 直到下一回合开始
        effects = append(effects, &SpellEffect{
            Type:      "buff",
            BuffType:  "ac",
            Value:     5,
            Duration:  "1 round",
        })

    case "bless":
        // 祝福术：攻击检定和豁免 +d4
        effects = append(effects, &SpellEffect{
            Type:         "buff",
            BuffType:     "dice",
            Value:        4,
            Duration:     "1 minute",
            Concentration: true,
        })
    }

    return effects
}
```

#### 3. 专注管理

```go
// internal/server/spell/concentration.go

type ConcentrationManager struct {
    stateManager *state.Manager
    dice         *dice.Service
}

// StartConcentration 开始专注
func (m *ConcentrationManager) StartConcentration(casterID, spellID string, duration string) error {
    caster, err := m.stateManager.GetCharacter(casterID)
    if err != nil {
        return err
    }

    if caster.Spellcasting == nil {
        return errors.New("character cannot cast spells")
    }

    // 结束之前的专注
    if caster.Spellcasting.Concentration != nil {
        m.EndConcentration(casterID)
    }

    caster.Spellcasting.Concentration = &ConcentrationInfo{
        SpellID:    spellID,
        StartedAt:  time.Now(),
        Duration:   duration,
    }

    return nil
}

// CheckConcentration 受伤时检查专注
func (m *ConcentrationManager) CheckConcentration(casterID string, damage int) (bool, error) {
    caster, err := m.stateManager.GetCharacter(casterID)
    if err != nil {
        return false, err
    }

    if caster.Spellcasting == nil || caster.Spellcasting.Concentration == nil {
        return true, nil // 没有专注中的法术
    }

    // DC = max(10, damage/2)
    dc := max(10, damage/2)

    // 体质豁免
    modifier := dice.GetModifier(caster.AbilityScores["constitution"])
    check := dice.AbilityCheck(modifier, dc, false, false)

    if !check.Success {
        // 专注失败，结束法术
        m.EndConcentration(casterID)
        return false, nil
    }

    return true, nil
}

// EndConcentration 结束专注
func (m *ConcentrationManager) EndConcentration(casterID string) error {
    caster, err := m.stateManager.GetCharacter(casterID)
    if err != nil {
        return err
    }

    if caster.Spellcasting != nil {
        caster.Spellcasting.Concentration = nil
    }

    return nil
}
```

```go
// internal/server/item/manager.go

package item

type Manager struct {
    itemData     *dnd5e.ItemData
    stateManager *state.Manager
}

// AddItem 添加物品到背包
func (m *Manager) AddItem(characterID, itemID string, count int) error {
    char, err := m.stateManager.GetCharacter(characterID)
    if err != nil {
        return err
    }

    // 检查负重
    item, err := m.itemData.Get(itemID)
    if err != nil {
        return err
    }

    currentWeight := m.calculateWeight(char)
    maxWeight := m.calculateMaxWeight(char)

    if currentWeight + (item.Weight * count) > maxWeight {
        return errors.New("exceeds carrying capacity")
    }

    // 查找是否已有该物品
    for _, invItem := range char.Inventory {
        if invItem.ItemID == itemID {
            invItem.Count += count
            return nil
        }
    }

    // 添加新物品
    char.Inventory = append(char.Inventory, &models.InventoryItem{
        ID:     generateID(),
        ItemID: itemID,
        Count:  count,
    })

    return nil
}

// Equip 装备物品
func (m *Manager) Equip(characterID, itemID, slot string) error {
    char, err := m.stateManager.GetCharacter(characterID)
    if err != nil {
        return err
    }

    // 查找物品
    var invItem *models.InventoryItem
    var invIndex int
    for i, item := range char.Inventory {
        if item.ID == itemID {
            invItem = item
            invIndex = i
            break
        }
    }

    if invItem == nil {
        return errors.New("item not found in inventory")
    }

    // 获取物品数据
    itemData, err := m.itemData.Get(invItem.ItemID)
    if err != nil {
        return err
    }

    // 检查是否可装备到此槽位
    if !m.canEquipToSlot(itemData, slot) {
        return errors.New("cannot equip to this slot")
    }

    // 如果槽位已有装备，先卸下
    if equippedID, exists := char.Equipment[slot]; exists && equippedID != "" {
        m.Unequip(characterID, slot)
    }

    // 装备
    char.Equipment[slot] = invItem.ItemID

    // 从背包移除（如果是单件物品）
    if invItem.Count == 1 {
        char.Inventory = append(char.Inventory[:invIndex], char.Inventory[invIndex+1:]...)
    } else {
        invItem.Count--
    }

    // 更新角色属性
    m.recalculateCharacterStats(char)

    return nil
}

// Unequip 卸下装备
func (m *Manager) Unequip(characterID, slot string) error {
    char, err := m.stateManager.GetCharacter(characterID)
    if err != nil {
        return err
    }

    itemID, exists := char.Equipment[slot]
    if !exists || itemID == "" {
        return errors.New("no item equipped in this slot")
    }

    // 添加回背包
    char.Inventory = append(char.Inventory, &models.InventoryItem{
        ID:     generateID(),
        ItemID: itemID,
        Count:  1,
    })

    // 清空槽位
    char.Equipment[slot] = ""

    // 更新角色属性
    m.recalculateCharacterStats(char)

    return nil
}

// UseItem 使用物品
func (m *Manager) UseItem(characterID, itemID string, targetID string) (*UseItemResult, error) {
    char, err := m.stateManager.GetCharacter(characterID)
    if err != nil {
        return nil, err
    }

    // 查找物品
    var invItem *models.InventoryItem
    var invIndex int
    for i, item := range char.Inventory {
        if item.ID == itemID {
            invItem = item
            invIndex = i
            break
        }
    }

    if invItem == nil {
        return nil, errors.New("item not found")
    }

    // 获取物品数据
    itemData, err := m.itemData.Get(invItem.ItemID)
    if err != nil {
        return nil, err
    }

    result := &UseItemResult{
        ItemID:   invItem.ItemID,
        ItemName: itemData.Name,
    }

    // 处理物品效果
    switch itemData.Type {
    case "potion":
        result.Effects = m.usePotion(char, itemData, targetID)
    case "scroll":
        result.Effects = m.useScroll(char, itemData, targetID)
    }

    // 消耗物品
    if invItem.Count == 1 {
        char.Inventory = append(char.Inventory[:invIndex], char.Inventory[invIndex+1:]...)
    } else {
        invItem.Count--
    }

    return result, nil
}

func (m *Manager) recalculateCharacterStats(char *models.Character) {
    // 基础 AC
    baseAC := 10
    dexMod := dice.GetModifier(char.AbilityScores["dexterity"])

    // 检查护甲
    if armorID, ok := char.Equipment["armor"]; ok && armorID != "" {
        armor, _ := m.itemData.GetArmor(armorID)
        if armor != nil {
            switch armor.Type {
            case "light":
                baseAC = armor.AC + dexMod
            case "medium":
                baseAC = armor.AC + min(dexMod, 2)
            case "heavy":
                baseAC = armor.AC
            }
        }
    }

    // 检查盾牌
    if shieldID, ok := char.Equipment["shield"]; ok && shieldID != "" {
        baseAC += 2
    }

    char.ArmorClass = baseAC
}
```

```go
// internal/client/tools/item_tool.go

func RegisterItemTools(registry *ToolRegistry, itemManager *item.Manager) {
    registry.Register(Tool{
        Name: "use_item",
        Description: "使用物品（药水、卷轴等）",
        InputSchema: map[string]interface{}{
            "type": "object",
            "properties": map[string]interface{}{
                "characterId": map[string]interface{}{"type": "string"},
                "itemId":      map[string]interface{}{"type": "string"},
                "targetId":    map[string]interface{}{"type": "string"},
            },
            "required": []string{"characterId", "itemId"},
        },
        Handler: func(args map[string]interface{}) (interface{}, error) {
            return itemManager.UseItem(
                args["characterId"].(string),
                args["itemId"].(string),
                getStringOr(args, "targetId", ""),
            )
        },
    })

    registry.Register(Tool{
        Name: "equip_item",
        Description: "装备物品到指定槽位",
        InputSchema: map[string]interface{}{
            "type": "object",
            "properties": map[string]interface{}{
                "characterId": map[string]interface{}{"type": "string"},
                "itemId":      map[string]interface{}{"type": "string"},
                "slot":        map[string]interface{}{"type": "string"},
            },
            "required": []string{"characterId", "itemId", "slot"},
        },
        Handler: func(args map[string]interface{}) (interface{}, error) {
            return nil, itemManager.Equip(
                args["characterId"].(string),
                args["itemId"].(string),
                args["slot"].(string),
            )
        },
    })
}
```
