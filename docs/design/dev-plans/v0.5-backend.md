#### 2. 剧本管理器

```go
// internal/server/scenario/manager.go

package scenario

import (
    "encoding/json"
    "os"
    "path/filepath"
    "dnd-game/backend/internal/shared/models"
    "dnd-game/backend/internal/shared/state"
)

type Manager struct {
    dataPath    string
    scenarios   map[string]*models.Scenario
    states      map[string]*ScenarioState // sessionID -> state
}

type ScenarioState struct {
    ScenarioID       string              `json:"scenarioId"`
    CurrentChapter   string              `json:"currentChapter"`
    CompletedChapters []string           `json:"completedChapters"`
    CurrentLocation  string              `json:"currentLocation"`
    VisitedLocations []string            `json:"visitedLocations"`
    UnlockedLocations []string           `json:"unlockedLocations"`
    Flags            map[string]bool     `json:"flags"`          // 剧情标志
    Variables        map[string]interface{} `json:"variables"`   // 剧情变量
}

func NewManager(dataPath string) *Manager {
    return &Manager{
        dataPath:  dataPath,
        scenarios: make(map[string]*models.Scenario),
        states:    make(map[string]*ScenarioState),
    }
}

// Load 加载剧本
func (m *Manager) Load(scenarioID string) (*models.Scenario, error) {
    // 检查缓存
    if s, ok := m.scenarios[scenarioID]; ok {
        return s, nil
    }

    // 从文件加载
    path := filepath.Join(m.dataPath, "scenarios", scenarioID, "scenario.json")
    data, err := os.ReadFile(path)
    if err != nil {
        return nil, err
    }

    var scenario models.Scenario
    if err := json.Unmarshal(data, &scenario); err != nil {
        return nil, err
    }

    m.scenarios[scenarioID] = &scenario
    return &scenario, nil
}

// Start 开始新剧本
func (m *Manager) Start(sessionID, scenarioID string) (*ScenarioState, error) {
    scenario, err := m.Load(scenarioID)
    if err != nil {
        return nil, err
    }

    state := &ScenarioState{
        ScenarioID:       scenarioID,
        CurrentChapter:   scenario.StartConfig.ChapterID,
        CompletedChapters: []string{},
        CurrentLocation:  scenario.StartConfig.LocationID,
        VisitedLocations: []string{scenario.StartConfig.LocationID},
        UnlockedLocations: []string{scenario.StartConfig.LocationID},
        Flags:            make(map[string]bool),
        Variables:        make(map[string]interface{}),
    }

    m.states[sessionID] = state
    return state, nil
}

// GetState 获取剧本状态
func (m *Manager) GetState(sessionID string) *ScenarioState {
    return m.states[sessionID]
}

// GetCurrentChapter 获取当前章节
func (m *Manager) GetCurrentChapter(sessionID string) (*models.Chapter, error) {
    state := m.GetState(sessionID)
    if state == nil {
        return nil, errors.New("no scenario state")
    }

    scenario, err := m.Load(state.ScenarioID)
    if err != nil {
        return nil, err
    }

    for _, chapter := range scenario.Chapters {
        if chapter.ID == state.CurrentChapter {
            return chapter, nil
        }
    }

    return nil, errors.New("chapter not found")
}

// AdvanceChapter 推进到下一章节
func (m *Manager) AdvanceChapter(sessionID string) error {
    state := m.GetState(sessionID)
    if state == nil {
        return errors.New("no scenario state")
    }

    // 标记当前章节完成
    state.CompletedChapters = append(state.CompletedChapters, state.CurrentChapter)

    // 查找下一章节
    scenario, err := m.Load(state.ScenarioID)
    if err != nil {
        return err
    }

    currentIndex := -1
    for i, chapter := range scenario.Chapters {
        if chapter.ID == state.CurrentChapter {
            currentIndex = i
            break
        }
    }

    if currentIndex >= 0 && currentIndex < len(scenario.Chapters)-1 {
        nextChapter := scenario.Chapters[currentIndex+1]

        // 检查解锁条件
        if m.checkConditions(state, nextChapter.UnlockConditions) {
            state.CurrentChapter = nextChapter.ID
            return nil
        }
    }

    return errors.New("no next chapter available")
}

// SetFlag 设置剧情标志
func (m *Manager) SetFlag(sessionID, flag string, value bool) {
    state := m.GetState(sessionID)
    if state != nil {
        state.Flags[flag] = value
    }
}

// GetFlag 获取剧情标志
func (m *Manager) GetFlag(sessionID, flag string) bool {
    state := m.GetState(sessionID)
    if state != nil {
        return state.Flags[flag]
    }
    return false
}

func (m *Manager) checkConditions(state *ScenarioState, conditions []Condition) bool {
    for _, cond := range conditions {
        switch cond.Type {
        case "flag":
            if state.Flags[cond.Key] != cond.Value.(bool) {
                return false
            }
        case "location_visited":
            found := false
            for _, loc := range state.VisitedLocations {
                if loc == cond.Key {
                    found = true
                    break
                }
            }
            if !found {
                return false
            }
        }
    }
    return true
}
```

```go
// internal/server/map/world_manager.go

package map

import (
    "dnd-game/backend/internal/shared/models"
    "dnd-game/backend/internal/server/scenario"
)

type WorldManager struct {
    scenarioManager *scenario.Manager
    combatManager   *combat.Manager
}

// Travel 执行旅行
func (m *WorldManager) Travel(sessionID, fromID, toID string) (*TravelResult, error) {
    state := m.scenarioManager.GetState(sessionID)
    if state == nil {
        return nil, errors.New("no scenario state")
    }

    scenario, err := m.scenarioManager.Load(state.ScenarioID)
    if err != nil {
        return nil, err
    }

    // 查找连接
    var connection *models.Connection
    for _, conn := range scenario.WorldMap.Connections {
        if (conn.From == fromID && conn.To == toID) ||
           (conn.From == toID && conn.To == fromID) {
            connection = conn
            break
        }
    }

    if connection == nil {
        return nil, errors.New("no path between locations")
    }

    // 检查目标是否解锁
    var targetLocation *models.Location
    for _, loc := range scenario.WorldMap.Locations {
        if loc.ID == toID {
            targetLocation = loc
            break
        }
    }

    if targetLocation == nil {
        return nil, errors.New("location not found")
    }

    if !targetLocation.Unlocked && !contains(state.UnlockedLocations, toID) {
        return nil, errors.New("location is locked")
    }

    result := &TravelResult{
        From:       fromID,
        To:         toID,
        TravelTime: connection.TravelTime,
    }

    // 检查是否触发随机遭遇
    if m.checkRandomEncounter(connection.DangerLevel) {
        encounter := m.selectRandomEncounter(targetLocation.Encounters)
        result.Encounter = encounter
        result.TriggeredCombat = true

        // 触发战斗
        m.combatManager.StartCombat(sessionID, encounter.EnemyIDs)
    }

    // 更新状态
    state.CurrentLocation = toID
    if !contains(state.VisitedLocations, toID) {
        state.VisitedLocations = append(state.VisitedLocations, toID)
    }

    return result, nil
}

// UnlockLocation 解锁地点
func (m *WorldManager) UnlockLocation(sessionID, locationID string) error {
    state := m.scenarioManager.GetState(sessionID)
    if state == nil {
        return errors.New("no scenario state")
    }

    if !contains(state.UnlockedLocations, locationID) {
        state.UnlockedLocations = append(state.UnlockedLocations, locationID)
    }

    return nil
}

// GetAvailableDestinations 获取可旅行的目的地
func (m *WorldManager) GetAvailableDestinations(sessionID string) ([]*models.Location, error) {
    state := m.scenarioManager.GetState(sessionID)
    if state == nil {
        return nil, errors.New("no scenario state")
    }

    scenario, err := m.scenarioManager.Load(state.ScenarioID)
    if err != nil {
        return nil, err
    }

    var destinations []*models.Location
    for _, conn := range scenario.WorldMap.Connections {
        var targetID string
        if conn.From == state.CurrentLocation {
            targetID = conn.To
        } else if conn.To == state.CurrentLocation {
            targetID = conn.From
        }

        if targetID != "" {
            for _, loc := range scenario.WorldMap.Locations {
                if loc.ID == targetID && contains(state.UnlockedLocations, targetID) {
                    destinations = append(destinations, loc)
                    break
                }
            }
        }
    }

    return destinations, nil
}

func (m *WorldManager) checkRandomEncounter(dangerLevel int) bool {
    // 危险等级越高，遭遇概率越大
    // 1: 10%, 2: 20%, 3: 30%, etc.
    chance := dangerLevel * 10
    roll := rand.Intn(100) + 1
    return roll <= chance
}

type TravelResult struct {
    From           string      `json:"from"`
    To             string      `json:"to"`
    TravelTime     int         `json:"travelTime"`
    Encounter      *Encounter  `json:"encounter,omitempty"`
    TriggeredCombat bool        `json:"triggeredCombat"`
}

type Encounter struct {
    ID         string   `json:"id"`
    Name       string   `json:"name"`
    EnemyIDs   []string `json:"enemyIds"`
    Difficulty string   `json:"difficulty"`
}
```

```go
// internal/persistence/save_manager.go

package persistence

import (
    "encoding/json"
    "os"
    "path/filepath"
    "time"
    "dnd-game/backend/internal/shared/state"
)

type SaveManager struct {
    savesPath string
}

type SaveMetadata struct {
    ID          string    `json:"id"`
    Name        string    `json:"name"`
    ScenarioID  string    `json:"scenarioId"`
    ChapterID   string    `json:"chapterId"`
    LocationID  string    `json:"locationId"`
    PartyLevel  int       `json:"partyLevel"`
    PlayTime    int       `json:"playTime"`    // 秒
    CreatedAt   time.Time `json:"createdAt"`
    UpdatedAt   time.Time `json:"updatedAt"`
}

func NewSaveManager(savesPath string) *SaveManager {
    os.MkdirAll(savesPath, 0755)
    return &SaveManager{savesPath: savesPath}
}

// List 列出所有存档
func (m *SaveManager) List() ([]*SaveMetadata, error) {
    entries, err := os.ReadDir(m.savesPath)
    if err != nil {
        return nil, err
    }

    var saves []*SaveMetadata
    for _, entry := range entries {
        if !entry.IsDir() {
            continue
        }

        metaPath := filepath.Join(m.savesPath, entry.Name(), "metadata.json")
        data, err := os.ReadFile(metaPath)
        if err != nil {
            continue
        }

        var meta SaveMetadata
        if err := json.Unmarshal(data, &meta); err != nil {
            continue
        }

        saves = append(saves, &meta)
    }

    // 按更新时间排序
    sort.Slice(saves, func(i, j int) bool {
        return saves[i].UpdatedAt.After(saves[j].UpdatedAt)
    })

    return saves, nil
}

// Create 创建存档
func (m *SaveManager) Create(name string, gameState *state.GameState) (*SaveMetadata, error) {
    saveID := generateSaveID()
    saveDir := filepath.Join(m.savesPath, saveID)

    if err := os.MkdirAll(saveDir, 0755); err != nil {
        return nil, err
    }

    // 计算队伍等级
    partyLevel := 1
    if len(gameState.Party) > 0 {
        partyLevel = gameState.Party[0].Level
    }

    meta := &SaveMetadata{
        ID:         saveID,
        Name:       name,
        ScenarioID: gameState.Scenario.ScenarioID,
        ChapterID:  gameState.Scenario.CurrentChapter,
        LocationID: gameState.Scenario.CurrentLocation,
        PartyLevel: partyLevel,
        PlayTime:   gameState.Metadata.PlayTime,
        CreatedAt:  time.Now(),
        UpdatedAt:  time.Now(),
    }

    // 保存元数据
    metaPath := filepath.Join(saveDir, "metadata.json")
    metaData, _ := json.MarshalIndent(meta, "", "  ")
    if err := os.WriteFile(metaPath, metaData, 0644); err != nil {
        return nil, err
    }

    // 保存游戏状态
    statePath := filepath.Join(saveDir, "state.json")
    stateData, _ := json.MarshalIndent(gameState, "", "  ")
    if err := os.WriteFile(statePath, stateData, 0644); err != nil {
        return nil, err
    }

    return meta, nil
}

// Load 加载存档
func (m *SaveManager) Load(saveID string) (*state.GameState, error) {
    statePath := filepath.Join(m.savesPath, saveID, "state.json")
    data, err := os.ReadFile(statePath)
    if err != nil {
        return nil, err
    }

    var gameState state.GameState
    if err := json.Unmarshal(data, &gameState); err != nil {
        return nil, err
    }

    return &gameState, nil
}

// Delete 删除存档
func (m *SaveManager) Delete(saveID string) error {
    saveDir := filepath.Join(m.savesPath, saveID)
    return os.RemoveAll(saveDir)
}

// QuickSave 快速存档
func (m *SaveManager) QuickSave(gameState *state.GameState) (*SaveMetadata, error) {
    name := fmt.Sprintf("自动存档 - %s", time.Now().Format("2006-01-02 15:04"))
    return m.Create(name, gameState)
}
```

```go
// internal/persistence/auto_save.go

package persistence

import (
    "time"
    "dnd-game/backend/internal/shared/state"
)

type AutoSaver struct {
    saveManager *SaveManager
    interval    time.Duration
    stopChan    chan struct{}
}

func NewAutoSaver(saveManager *SaveManager, intervalSeconds int) *AutoSaver {
    return &AutoSaver{
        saveManager: saveManager,
        interval:    time.Duration(intervalSeconds) * time.Second,
        stopChan:    make(chan struct{}),
    }
}

func (a *AutoSaver) Start(getState func() *state.GameState) {
    ticker := time.NewTicker(a.interval)
    defer ticker.Stop()

    for {
        select {
        case <-ticker.C:
            gameState := getState()
            if gameState != nil {
                a.saveManager.QuickSave(gameState)
            }
        case <-a.stopChan:
            return
        }
    }
}

func (a *AutoSaver) Stop() {
    close(a.stopChan)
}
```
