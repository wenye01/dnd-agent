```typescript
// src/game/scenes/WorldMapScene.ts

import Phaser from 'phaser'
import { useGameStore } from '../../stores/gameStore'
import { useWebSocket } from '../../contexts/WebSocketContext'

export class WorldMapScene extends Phaser.Scene {
  private mapImage!: Phaser.GameObjects.Image
  private locationMarkers: Map<string, Phaser.GameObjects.Container> = new Map()
  private travelLine!: Phaser.GameObjects.Graphics

  constructor() {
    super({ key: 'worldMap' })
  }

  create() {
    // 绘制背景
    this.createBackground()

    // 加载地点
    this.createLocations()

    // 设置交互
    this.setupInteraction()

    // 监听更新
    this.listenForUpdates()
  }

  private createBackground() {
    const width = this.cameras.main.width
    const height = this.cameras.main.height

    // 创建简单的背景
    const graphics = this.add.graphics()

    // 背景
    graphics.fillStyle(0x1a1a2e, 1)
    graphics.fillRect(0, 0, width, height)

    // 添加网格
    graphics.lineStyle(1, 0x333366, 0.3)
    for (let x = 0; x < width; x += 50) {
      graphics.moveTo(x, 0)
      graphics.lineTo(x, height)
    }
    for (let y = 0; y < height; y += 50) {
      graphics.moveTo(0, y)
      graphics.lineTo(width, y)
    }
    graphics.strokePath()
  }

  private createLocations() {
    const gameState = useGameStore.getState().gameState
    if (!gameState?.scenario?.worldMap) return

    const { locations } = gameState.scenario.worldMap
    const currentLocation = gameState.scenario.currentLocation

    locations.forEach((loc) => {
      const marker = this.createLocationMarker(loc, loc.id === currentLocation)
      this.locationMarkers.set(loc.id, marker)
    })
  }

  private createLocationMarker(location: Location, isCurrent: boolean): Phaser.GameObjects.Container {
    const x = location.position.x
    const y = location.position.y

    // 创建容器
    const container = this.add.container(x, y)

    // 创建标记点
    const circle = this.add.graphics()
    if (location.unlocked) {
      circle.fillStyle(isCurrent ? 0x44ff44 : 0x4488ff, 1)
    } else {
      circle.fillStyle(0x666666, 0.5)
    }
    circle.fillCircle(0, 0, 20)

    // 创建名称文本
    const text = this.add.text(0, 30, location.name, {
      fontSize: '14px',
      color: location.unlocked ? '#ffffff' : '#666666',
    })
    text.setOrigin(0.5)

    // 创建图标
    const icon = this.add.text(0, 0, this.getLocationIcon(location.type), {
      fontSize: '20px',
    })
    icon.setOrigin(0.5)

    container.add([circle, icon, text])

    // 设置交互
    if (location.unlocked) {
      container.setSize(40, 40)
      container.setInteractive()

      container.on('pointerover', () => {
        container.setScale(1.2)
        this.showLocationTooltip(location)
      })

      container.on('pointerout', () => {
        container.setScale(1)
        this.hideTooltip()
      })

      container.on('pointerdown', () => {
        this.onLocationClick(location)
      })
    }

    return container
  }

  private getLocationIcon(type: string): string {
    const icons: Record<string, string> = {
      city: '🏛️',
      village: '🏠',
      dungeon: '🏰',
      forest: '🌲',
      mountain: '⛰️',
      lake: '💧',
      ruins: '🏚️',
    }
    return icons[type] || '📍'
  }

  private showLocationTooltip(location: Location) {
    // 创建提示框
    const tooltip = this.add.container(0, 0)

    const bg = this.add.graphics()
    bg.fillStyle(0x000000, 0.8)
    bg.fillRoundedRect(0, 0, 200, 80, 8)

    const title = this.add.text(10, 10, location.name, {
      fontSize: '16px',
      color: '#ffffff',
      fontStyle: 'bold',
    })

    const desc = this.add.text(10, 35, location.description, {
      fontSize: '12px',
      color: '#cccccc',
      wordWrap: { width: 180 },
    })

    tooltip.add([bg, title, desc])
    tooltip.setPosition(location.position.x + 30, location.position.y - 40)
    tooltip.setName('tooltip')

    // 移除旧提示
    this.hideTooltip()
  }

  private hideTooltip() {
    const oldTooltip = this.children.getByName('tooltip')
    if (oldTooltip) {
      oldTooltip.destroy()
    }
  }

  private onLocationClick(location: Location) {
    const gameState = useGameStore.getState().gameState
    const currentLocation = gameState?.scenario?.currentLocation

    if (location.id === currentLocation) {
      // 当前位置，切换到场景地图
      this.scene.start('sceneMap')
      return
    }

    // 显示旅行确认
    this.showTravelConfirm(currentLocation, location)
  }

  private showTravelConfirm(fromId: string, to: Location) {
    const from = this.getLocationById(fromId)

    // 计算旅行时间
    const travelTime = this.calculateTravelTime(from, to)

    // 显示确认对话框
    const confirmDialog = document.createElement('div')
    confirmDialog.className = 'travel-confirm'
    confirmDialog.innerHTML = `
      <div class="travel-content">
        <h3>确认旅行</h3>
        <p>从 ${from.name} 到 ${to.name}</p>
        <p>预计时间: ${travelTime} 小时</p>
        <div class="travel-buttons">
          <button id="confirm-travel">确认</button>
          <button id="cancel-travel">取消</button>
        </div>
      </div>
    `

    document.body.appendChild(confirmDialog)

    document.getElementById('confirm-travel')?.addEventListener('click', () => {
      this.executeTravel(from, to)
      confirmDialog.remove()
    })

    document.getElementById('cancel-travel')?.addEventListener('click', () => {
      confirmDialog.remove()
    })
  }

  private executeTravel(from: Location, to: Location) {
    const { send } = useWebSocket()

    // 绘制旅行路径
    this.drawTravelPath(from.position, to.position)

    // 发送旅行请求
    send({
      type: 'map_action',
      payload: {
        action: 'travel',
        from: from.id,
        to: to.id,
      },
    })
  }

  private drawTravelPath(from: Position, to: Position) {
    this.travelLine.clear()
    this.travelLine.lineStyle(3, 0xffff00, 1)
    this.travelLine.moveTo(from.x, from.y)
    this.travelLine.lineTo(to.x, to.y)
    this.travelLine.strokePath()

    // 播放动画
    this.tweens.add({
      targets: this.travelLine,
      alpha: 0,
      duration: 2000,
    })
  }

  private listenForUpdates() {
    const { subscribe } = useWebSocket()

    subscribe((message) => {
      if (message.type === 'state_update') {
        const payload = message.payload as any
        if (payload.stateType === 'location') {
          this.updateLocations()
        }
      }
    })
  }
}
```

#### 1. 存档对话框

```tsx
// src/components/dialogs/SaveLoadDialog.tsx

import { useState, useEffect } from 'react'
import { useWebSocket } from '../../contexts/WebSocketContext'
import { Dialog } from '../ui/Dialog'
import { Button } from '../ui/Button'

interface SaveMetadata {
  id: string
  name: string
  scenarioId: string
  chapterId: string
  locationId: string
  partyLevel: number
  playTime: number
  createdAt: string
  updatedAt: string
}

export function SaveLoadDialog({ onClose }: { onClose: () => void }) {
  const [saves, setSaves] = useState<SaveMetadata[]>([])
  const [loading, setLoading] = useState(false)
  const [selectedSave, setSelectedSave] = useState<string | null>(null)
  const [newSaveName, setNewSaveName] = useState('')
  const [mode, setMode] = useState<'load' | 'save'>('load')

  const { send, subscribe } = useWebSocket()

  useEffect(() => {
    // 获取存档列表
    send({
      type: 'management',
      payload: { operation: 'list_saves' },
    })

    return subscribe((message) => {
      if (message.type === 'state_update') {
        const payload = message.payload as any
        if (payload.stateType === 'saves') {
          setSaves(payload.data)
          setLoading(false)
        }
      }
    })
  }, [])

  const handleCreateSave = () => {
    if (!newSaveName.trim()) return

    send({
      type: 'management',
      payload: {
        operation: 'create_save',
        params: { name: newSaveName.trim() },
      },
    })
    setNewSaveName('')
    onClose()
  }

  const handleLoadSave = () => {
    if (!selectedSave) return

    send({
      type: 'management',
      payload: {
        operation: 'load_save',
        params: { saveId: selectedSave },
      },
    })
    onClose()
  }

  const handleDeleteSave = () => {
    if (!selectedSave) return

    if (confirm('确定要删除此存档吗？此操作不可撤销。')) {
      send({
        type: 'management',
        payload: {
          operation: 'delete_save',
          params: { saveId: selectedSave },
        },
      })
      setSelectedSave(null)
    }
  }

  const formatPlayTime = (seconds: number) => {
    const hours = Math.floor(seconds / 3600)
    const minutes = Math.floor((seconds % 3600) / 60)
    return `${hours}小时${minutes}分钟`
  }

  return (
    <Dialog title="存档管理" onClose={onClose}>
      {/* 模式切换 */}
      <div className="flex border-b mb-4">
        <button
          onClick={() => setMode('load')}
          className={`flex-1 py-2 text-center ${
            mode === 'load' ? 'border-b-2 border-primary-500 font-bold' : ''
          }`}
        >
          读取存档
        </button>
        <button
          onClick={() => setMode('save')}
          className={`flex-1 py-2 text-center ${
            mode === 'save' ? 'border-b-2 border-primary-500 font-bold' : ''
          }`}
        >
          创建存档
        </button>
      </div>

      {mode === 'save' ? (
        // 创建存档
        <div className="space-y-4">
          <div>
            <label className="block text-sm font-bold mb-2">存档名称</label>
            <input
              type="text"
              value={newSaveName}
              onChange={(e) => setNewSaveName(e.target.value)}
              placeholder="输入存档名称..."
              className="w-full px-3 py-2 border rounded"
            />
          </div>
          <Button
            onClick={handleCreateSave}
            disabled={!newSaveName.trim()}
            className="w-full"
          >
            创建存档
          </Button>
        </div>
      ) : (
        // 读取存档
        <>
          {loading ? (
            <div className="text-center py-8">加载中...</div>
          ) : saves.length === 0 ? (
            <div className="text-center py-8 text-gray-500">暂无存档</div>
          ) : (
            <div className="space-y-2 max-h-60 overflow-y-auto">
              {saves.map((save) => (
                <div
                  key={save.id}
                  onClick={() => setSelectedSave(save.id)}
                  className={`
                    p-3 rounded border cursor-pointer
                    ${selectedSave === save.id ? 'border-primary-500 bg-primary-50' : 'hover:bg-gray-50'}
                  `}
                >
                  <div className="flex justify-between">
                    <span className="font-bold">{save.name}</span>
                    <span className="text-sm text-gray-500">
                      Lv.{save.partyLevel}
                    </span>
                  </div>
                  <div className="text-sm text-gray-600 mt-1">
                    {save.scenarioId} - {save.chapterId}
                  </div>
                  <div className="flex justify-between text-xs text-gray-400 mt-1">
                    <span>{formatPlayTime(save.playTime)}</span>
                    <span>{new Date(save.updatedAt).toLocaleDateString()}</span>
                  </div>
                </div>
              ))}
            </div>
          )}

          <div className="flex gap-2 mt-4">
            <Button
              onClick={handleLoadSave}
              disabled={!selectedSave}
              className="flex-1"
            >
              读取
            </Button>
            <Button
              onClick={handleDeleteSave}
              variant="danger"
              disabled={!selectedSave}
              className="flex-1"
            >
              删除
            </Button>
          </div>
        </>
      )}
    </Dialog>
  )
}
```

```tsx
// src/components/panels/ChatHistoryPanel.tsx

import { useState, useEffect, useRef } from 'react'
import { useChatStore } from '../../stores/chatStore'
import { Panel } from '../ui/Panel'
import { Input } from '../ui/Input'

export function ChatHistoryPanel() {
  const messages = useChatStore((s) => s.messages)
  const [searchText, setSearchText] = useState('')
  const [filter, setFilter] = useState<'all' | 'user' | 'dm'>('all')
  const [visibleCount, setVisibleCount] = useState(50)
  const listRef = useRef<HTMLDivElement>(null)

  // 过滤消息
  const filteredMessages = messages.filter((msg) => {
    if (filter !== 'all' && msg.type !== filter) {
      return false
    }
    if (searchText && !msg.content.toLowerCase().includes(searchText.toLowerCase())) {
      return false
    }
    return true
  })

  // 显示的消息（倒序，最新的在上面）
  const displayedMessages = filteredMessages.slice(-visibleCount).reverse()

  // 滚动加载更多
  const handleScroll = () => {
    if (!listRef.current) return
    const { scrollTop, scrollHeight } = listRef.current
    if (scrollTop === 0 && visibleCount < filteredMessages.length) {
      setVisibleCount((c) => Math.min(c + 50, filteredMessages.length))
    }
  }

  // 格式化时间
  const formatTime = (timestamp: number) => {
    const date = new Date(timestamp)
    return date.toLocaleTimeString('zh-CN', {
      hour: '2-digit',
      minute: '2-digit',
    })
  }

  return (
    <Panel title="对话历史">
      {/* 搜索和过滤 */}
      <div className="mb-3 space-y-2">
        <Input
          value={searchText}
          onChange={(e) => setSearchText(e.target.value)}
          placeholder="搜索..."
        />

        <div className="flex gap-2">
          <button
            onClick={() => setFilter('all')}
            className={`px-2 py-1 text-xs rounded ${
              filter === 'all' ? 'bg-primary-500 text-white' : 'bg-gray-200'
            }`}
          >
            全部
          </button>
          <button
            onClick={() => setFilter('user')}
            className={`px-2 py-1 text-xs rounded ${
              filter === 'user' ? 'bg-primary-500 text-white' : 'bg-gray-200'
            }`}
          >
            我的输入
          </button>
          <button
            onClick={() => setFilter('dm')}
            className={`px-2 py-1 text-xs rounded ${
              filter === 'dm' ? 'bg-primary-500 text-white' : 'bg-gray-200'
            }`}
          >
            DM 叙述
          </button>
        </div>
      </div>

      {/* 统计 */}
      <div className="text-xs text-gray-500 mb-2">
        共 {filteredMessages.length} 条消息
      </div>

      {/* 消息列表 */}
      <div
        ref={listRef}
        onScroll={handleScroll}
        className="space-y-2 max-h-96 overflow-y-auto"
      >
        {displayedMessages.map((msg) => (
          <div
            key={msg.id}
            className={`
              p-2 rounded text-sm
              ${msg.type === 'user' ? 'bg-primary-50 border-l-2 border-primary-500' : 'bg-gray-50'}
            `}
          >
            <div className="flex justify-between text-xs text-gray-500 mb-1">
              <span className="font-bold">
                {msg.type === 'user' ? '我' : 'DM'}
              </span>
              <span>{formatTime(msg.timestamp)}</span>
            </div>
            <div className="whitespace-pre-wrap">{msg.content}</div>
          </div>
        ))}

        {visibleCount < filteredMessages.length && (
          <div className="text-center text-xs text-gray-400 py-2">
            向上滚动加载更多...
          </div>
        )}
      </div>
    </Panel>
  )
}
```
