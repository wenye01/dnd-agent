#### 1. 死亡豁免管理器

```go
// internal/server/character/death.go

package character

import (
    "dnd-game/backend/internal/server/dice"
    "dnd-game/backend/internal/shared/models"
)

type DeathManager struct {
    diceRoller *dice.Service
}

type DeathSaveResult struct {
    Roll        int    `json:"roll"`
    IsCritical  bool   `json:"isCritical"`  // 1 或 20
    Successes   int    `json:"successes"`
    Failures    int    `json:"failures"`
    State       string `json:"state"`       // alive, dying, stable, dead
}

func (d *DeathManager) MakeDeathSave(character *models.Character) *DeathSaveResult {
    result := &DeathSaveResult{}

    // 掷 d20
    roll := d.diceRoller.RollD20()
    result.Roll = roll

    // 初始化死亡豁免计数
    if character.DeathSaves == nil {
        character.DeathSaves = &models.DeathSaves{}
    }

    // 处理暴击
    if roll == 20 {
        // 暴击成功：恢复 1 HP
        character.CurrentHitPoints = 1
        character.DeathSaves = nil
        result.State = "alive"
        result.IsCritical = true
        return result
    }

    if roll == 1 {
        // 暴击失败：算两次失败
        character.DeathSaves.Failures += 2
        result.IsCritical = true
    } else if roll >= 10 {
        // 成功
        character.DeathSaves.Successes++
    } else {
        // 失败
        character.DeathSaves.Failures++
    }

    result.Successes = character.DeathSaves.Successes
    result.Failures = character.DeathSaves.Failures

    // 检查状态
    if character.DeathSaves.Successes >= 3 {
        character.DeathSaves = nil
        result.State = "stable"
    } else if character.DeathSaves.Failures >= 3 {
        result.State = "dead"
    } else {
        result.State = "dying"
    }

    return result
}

// ApplyDamageToDying 对濒死角色造成伤害
func (d *DeathManager) ApplyDamageToDying(character *models.Character, isCritical bool) {
    if character.DeathSaves == nil {
        character.DeathSaves = &models.DeathSaves{}
    }

    if isCritical {
        // 暴击伤害：两次失败
        character.DeathSaves.Failures += 2
    } else {
        character.DeathSaves.Failures++
    }

    // 检查是否死亡
    if character.DeathSaves.Failures >= 3 {
        // 角色死亡
    }
}

// Stabilize 稳定濒死角色
func (d *DeathManager) Stabilize(character *models.Character) {
    character.DeathSaves = nil
    // HP 仍然是 0，但不再进行死亡豁免
}

// IsDying 检查角色是否濒死
func (d *DeathManager) IsDying(character *models.Character) bool {
    return character.CurrentHitPoints <= 0 && character.DeathSaves != nil
}

// IsDead 检查角色是否死亡
func (d *DeathManager) IsDead(character *models.Character) bool {
    return character.CurrentHitPoints <= 0 && character.DeathSaves == nil && !containsCondition(character, "unconscious")
}
```

```go
// internal/server/spell/resurrection.go

package spell

import (
    "time"
    "dnd-game/backend/internal/shared/models"
)

type ResurrectionManager struct {
    spellData *dnd5e.SpellData
}

type ResurrectionResult struct {
    Success    bool   `json:"success"`
    Message    string `json:"message"`
    Penalties  []string `json:"penalties"`
}

// CanResurrect 检查是否可以复活
func (r *ResurrectionManager) CanResurrect(target *models.Character, spellID string) (bool, string) {
    spell, _ := r.spellData.Get(spellID)

    // 检查死亡时间
    deathTime := target.DeathTime
    if deathTime == nil {
        return false, "角色未死亡"
    }

    // 根据法术检查时间限制
    maxDays := r.getMaxDays(spellID)
    daysSinceDeath := time.Since(*deathTime).Hours() / 24

    if daysSinceDeath > float64(maxDays) {
        return false, fmt.Sprintf("死亡时间超过 %d 天，无法复活", maxDays)
    }

    // 检查尸体完整性（某些法术需要）
    if r.requiresBody(spellID) && !target.HasBody {
        return false, "没有完整的尸体"
    }

    return true, ""
}

// Resurrect 执行复活
func (r *ResurrectionManager) Resurrect(target *models.Character, spellID string) *ResurrectionResult {
    canResurrect, reason := r.CanResurrect(target, spellID)
    if !canResurrect {
        return &ResurrectionResult{
            Success: false,
            Message: reason,
        }
    }

    result := &ResurrectionResult{
        Success: true,
        Message: "复活成功",
    }

    // 恢复 HP
    target.CurrentHitPoints = 1
    target.DeathSaves = nil
    target.DeathTime = nil

    // 移除死亡相关状态
    target.Conditions = removeCondition(target.Conditions, "unconscious")

    // 应用复活惩罚
    switch spellID {
    case "raise_dead":
        result.Penalties = append(result.Penalties, "攻击检定和豁免 -4（持续 5 天）")
        target.AddTemporaryModifier("attack", -4, 5*24*time.Hour)
        target.AddTemporaryModifier("savingThrows", -4, 5*24*time.Hour)
    case "reincarnate":
        // 可能改变种族
        newRace := r.rollForNewRace()
        result.Penalties = append(result.Penalties, fmt.Sprintf("种族变为 %s", newRace))
        target.Race = newRace
    }

    return result
}

func (r *ResurrectionManager) getMaxDays(spellID string) int {
    switch spellID {
    case "revivify":
        return 0 // 1 分钟内
    case "raise_dead":
        return 10
    case "reincarnate":
        return 10
    case "resurrection":
        return 100
    case "true_resurrection":
        return 200
    default:
        return 10
    }
}

func (r *ResurrectionManager) requiresBody(spellID string) bool {
    return spellID != "true_resurrection"
}
```

```go
package npc

type ResurrectionService struct {
    costs map[string]int
}

func (s *ResurrectionService) OfferResurrection(character *models.Character) *ResurrectionOffer {
    // 根据角色等级计算费用
    baseCost := 1000 * character.Level

    return &ResurrectionOffer{
        Available: true,
        Cost:      baseCost,
        Service:   "复活术",
        Location:  "城镇神殿",
    }
}

func (s *ResurrectionService) PerformResurrection(character *models.Character, gold int) (*ResurrectionResult, error) {
    offer := s.OfferResurrection(character)

    if gold < offer.Cost {
        return nil, fmt.Errorf("金币不足，需要 %d", offer.Cost)
    }

    // 扣除金币
    character.Gold -= offer.Cost

    // 执行复活（简化版）
    character.CurrentHitPoints = 1
    character.DeathSaves = nil
    character.DeathTime = nil

    return &ResurrectionResult{
        Success: true,
        Message: fmt.Sprintf("支付 %d 金币后复活成功", offer.Cost),
    }, nil
}
```

```go
// internal/shared/state/snapshot.go

package state

import (
    "encoding/json"
    "sync"
)

type SnapshotManager struct {
    mu        sync.RWMutex
    snapshots map[string]*GameState
    maxSnaps  int
}

func NewSnapshotManager(maxSnapshots int) *SnapshotManager {
    return &SnapshotManager{
        snapshots: make(map[string]*GameState),
        maxSnaps:  maxSnapshots,
    }
}

// CreateSnapshot 创建快照
func (sm *SnapshotManager) CreateSnapshot(gameState *GameState) string {
    sm.mu.Lock()
    defer sm.mu.Unlock()

    // 深拷贝
    data, _ := json.Marshal(gameState)
    var copy GameState
    json.Unmarshal(data, &copy)

    snapshotID := generateID()
    sm.snapshots[snapshotID] = &copy

    // 清理旧快照
    if len(sm.snapshots) > sm.maxSnaps {
        sm.cleanOldSnapshots()
    }

    return snapshotID
}

// RestoreSnapshot 恢复快照
func (sm *SnapshotManager) RestoreSnapshot(snapshotID string) (*GameState, error) {
    sm.mu.RLock()
    defer sm.mu.RUnlock()

    snapshot, ok := sm.snapshots[snapshotID]
    if !ok {
        return nil, fmt.Errorf("snapshot not found: %s", snapshotID)
    }

    // 返回快照的拷贝
    data, _ := json.Marshal(snapshot)
    var copy GameState
    json.Unmarshal(data, &copy)

    return &copy, nil
}

// ExecuteWithRollback 执行操作，失败时回滚
func (sm *SnapshotManager) ExecuteWithRollback(
    gameState *GameState,
    operation func() error,
) error {
    // 创建快照
    snapshotID := sm.CreateSnapshot(gameState)

    // 执行操作
    if err := operation(); err != nil {
        // 回滚
        restored, restoreErr := sm.RestoreSnapshot(snapshotID)
        if restoreErr != nil {
            return fmt.Errorf("operation failed: %v, rollback also failed: %v", err, restoreErr)
        }

        // 恢复状态
        *gameState = *restored
        return err
    }

    return nil
}
```

#### 2. LLM 调用重试

```go
// internal/client/llm/retry.go

package llm

import (
    "context"
    "time"
)

type RetryConfig struct {
    MaxAttempts int
    BaseDelay   time.Duration
    MaxDelay    time.Duration
    Multiplier  float64
}

func DefaultRetryConfig() *RetryConfig {
    return &RetryConfig{
        MaxAttempts: 3,
        BaseDelay:   1 * time.Second,
        MaxDelay:    30 * time.Second,
        Multiplier:  2.0,
    }
}

// RetryWithBackoff 带退避的重试
func (p *OpenAIProvider) RetryWithBackoff(
    ctx context.Context,
    req *Request,
    config *RetryConfig,
) (*Response, error) {
    var lastErr error
    delay := config.BaseDelay

    for attempt := 1; attempt <= config.MaxAttempts; attempt++ {
        resp, err := p.SendMessage(ctx, req)
        if err == nil {
            return resp, nil
        }

        lastErr = err

        // 检查是否可重试
        if !isRetryableError(err) {
            return nil, err
        }

        // 等待后重试
        select {
        case <-ctx.Done():
            return nil, ctx.Err()
        case <-time.After(delay):
        }

        // 指数退避
        delay = time.Duration(float64(delay) * config.Multiplier)
        if delay > config.MaxDelay {
            delay = config.MaxDelay
        }
    }

    return nil, fmt.Errorf("max retry attempts reached: %w", lastErr)
}

func isRetryableError(err error) bool {
    // 检查是否是可重试的错误
    // - 网络超时
    // - 服务暂时不可用 (503)
    // - 速率限制 (429)
    return true
}
```

```go
// internal/persistence/error_log.go

package persistence

import (
    "encoding/json"
    "os"
    "path/filepath"
    "time"
)

type ErrorLog struct {
    Timestamp   time.Time              `json:"timestamp"`
    Type        string                 `json:"type"`
    Message     string                 `json:"message"`
    StackTrace  string                 `json:"stackTrace,omitempty"`
    Context     map[string]interface{} `json:"context,omitempty"`
    Recovered   bool                   `json:"recovered"`
    RecoveryAction string              `json:"recoveryAction,omitempty"`
}

type ErrorLogger struct {
    logPath string
}

func (l *ErrorLogger) Log(err error, context map[string]interface{}, recovered bool, action string) {
    log := ErrorLog{
        Timestamp:      time.Now(),
        Type:           fmt.Sprintf("%T", err),
        Message:        err.Error(),
        StackTrace:     getStackTrace(),
        Context:        context,
        Recovered:      recovered,
        RecoveryAction: action,
    }

    data, _ := json.Marshal(log)
    line := append(data, '\n')

    path := filepath.Join(l.logPath, "errors.jsonl")
    f, _ := os.OpenFile(path, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0644)
    defer f.Close()
    f.Write(line)
}
```

```go
// internal/api/rest/admin.go

package rest

import (
    "github.com/gin-gonic/gin"
    "dnd-game/backend/internal/shared/state"
)

type AdminHandler struct {
    stateManager *state.Manager
    combatManager *combat.Manager
    characterManager *character.Manager
}

func RegisterAdminRoutes(router *gin.Engine, handler *AdminHandler) {
    admin := router.Group("/api/admin")
    admin.Use(requireAdminAuth()) // 可选的管理员认证

    admin.POST("/combat/end", handler.ForceEndCombat)
    admin.POST("/character/:id/modify", handler.ModifyCharacter)
    admin.POST("/state/reset", handler.ResetState)
    admin.GET("/debug/state", handler.GetDebugState)
    admin.GET("/debug/logs", handler.GetRecentLogs)
}

// ForceEndCombat 强制结束战斗
func (h *AdminHandler) ForceEndCombat(c *gin.Context) {
    gameState := h.stateManager.GetGameState()
    if gameState == nil || gameState.Combat == nil {
        c.JSON(400, gin.H{"error": "no active combat"})
        return
    }

    // 强制结束战斗
    h.combatManager.ForceEnd()

    c.JSON(200, gin.H{
        "success": true,
        "message": "combat force ended",
    })
}

// ModifyCharacter 修改角色数据
func (h *AdminHandler) ModifyCharacter(c *gin.Context) {
    characterID := c.Param("id")

    var req struct {
        Field string      `json:"field"`
        Value interface{} `json:"value"`
    }

    if err := c.BindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }

    character, err := h.characterManager.GetCharacter(characterID)
    if err != nil {
        c.JSON(404, gin.H{"error": "character not found"})
        return
    }

    // 修改字段
    switch req.Field {
    case "currentHitPoints":
        character.CurrentHitPoints = int(req.Value.(float64))
    case "gold":
        character.Gold = int(req.Value.(float64))
    case "experiencePoints":
        character.ExperiencePoints = int(req.Value.(float64))
    default:
        c.JSON(400, gin.H{"error": "unsupported field"})
        return
    }

    c.JSON(200, gin.H{
        "success": true,
        "character": character,
    })
}

// ResetState 重置状态
func (h *AdminHandler) ResetState(c *gin.Context) {
    var req struct {
        Type string `json:"type"` // all, combat, party, scenario
    }

    if err := c.BindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }

    switch req.Type {
    case "combat":
        h.stateManager.UpdateGameState(func(s *state.GameState) {
            s.Combat = nil
        })
    case "party":
        h.stateManager.UpdateGameState(func(s *state.GameState) {
            s.Party = nil
        })
    case "all":
        h.stateManager.Reset()
    }

    c.JSON(200, gin.H{
        "success": true,
        "message": fmt.Sprintf("%s reset", req.Type),
    })
}

// GetDebugState 获取调试状态
func (h *AdminHandler) GetDebugState(c *gin.Context) {
    gameState := h.stateManager.GetGameState()

    c.JSON(200, gin.H{
        "gameState": gameState,
        "timestamp": time.Now().Unix(),
    })
}
```
