```tsx
// src/components/panels/CharacterPanel.tsx

import { useGameStore } from '../../stores/gameStore'
import { Panel } from '../ui/Panel'
import { AbilityScores } from './AbilityScores'
import { EquipmentSlots } from './EquipmentSlots'
import { SkillsList } from './SkillsList'
import { ConditionsList } from './ConditionsList'
import { SpellSlotsBar } from './SpellSlotsBar'

export function CharacterPanel() {
  const party = useGameStore((s) => s.gameState?.party)
  const selectedId = useUIStore((s) => s.selectedCharacterId)
  const setSelectedId = useUIStore((s) => s.setSelectedCharacterId)

  const selectedChar = party?.find((c) => c.id === selectedId) || party?.[0]

  if (!party || party.length === 0) {
    return (
      <Panel title="角色">
        <p className="text-gray-500 text-sm">尚未创建角色</p>
      </Panel>
    )
  }

  return (
    <div className="p-4 space-y-4 overflow-y-auto h-full">
      {/* 角色选择 */}
      {party.length > 1 && (
        <div className="flex gap-2">
          {party.map((char) => (
            <button
              key={char.id}
              onClick={() => setSelectedId(char.id)}
              className={`
                px-3 py-1 rounded text-sm
                ${char.id === selectedId
                  ? 'bg-primary-500 text-white'
                  : 'bg-gray-200 text-gray-700'}
              `}
            >
              {char.name}
            </button>
          ))}
        </div>
      )}

      {/* 选中角色详情 */}
      {selectedChar && (
        <>
          {/* 头部信息 */}
          <Panel title={selectedChar.name}>
            <div className="space-y-2">
              <div className="text-sm text-gray-600">
                {selectedChar.race} {selectedChar.class} Lv.{selectedChar.level}
              </div>
              <div className="text-xs text-gray-500">
                {selectedChar.background} / {selectedChar.alignment}
              </div>
            </div>
          </Panel>

          {/* HP 条 */}
          <HealthBar
            current={selectedChar.currentHitPoints}
            max={selectedChar.maxHitPoints}
            temp={selectedChar.temporaryHitPoints}
          />

          {/* 法术位（如果有） */}
          {selectedChar.spellcasting && (
            <SpellSlotsBar slots={selectedChar.spellcasting.spellSlots} />
          )}

          {/* 状态效果 */}
          {selectedChar.conditions.length > 0 && (
            <ConditionsList conditions={selectedChar.conditions} />
          )}

          {/* 属性 */}
          <AbilityScores scores={selectedChar.abilityScores} />

          {/* 装备 */}
          <EquipmentSlots equipment={selectedChar.equipment} />

          {/* 技能 */}
          <SkillsList skills={selectedChar.skills} />
        </>
      )}
    </div>
  )
}
```

// src/components/panels/HealthBar.tsx

interface HealthBarProps {
  current: number
  max: number
  temp?: number
}

export function HealthBar({ current, max, temp = 0 }: HealthBarProps) {
  const percentage = Math.max(0, Math.min(100, (current / max) * 100))

  const getColor = () => {
    if (percentage > 50) return 'bg-green-500'
    if (percentage > 25) return 'bg-yellow-500'
    return 'bg-red-500'
  }

  return (
    <div className="space-y-1">
      <div className="flex justify-between text-sm">
        <span className="font-bold">HP</span>
        <span>
          {current}/{max}
          {temp > 0 && <span className="text-blue-500"> (+{temp})</span>}
        </span>
      </div>
      <div className="h-4 bg-gray-200 rounded overflow-hidden">
        {/* 临时 HP */}
        {temp > 0 && (
          <div
            className="h-full bg-blue-400"
            style={{ width: `${(temp / max) * 100}%` }}
          />
        )}
        {/* 当前 HP */}
        <div
          className={`h-full ${getColor()} transition-all`}
          style={{ width: `${percentage}%` }}
        />
      </div>
    </div>
  )
}

// src/components/panels/AbilityScores.tsx

import type { AbilityScores as AbilityScoresType } from '../../types'

interface AbilityScoresProps {
  scores: AbilityScoresType
}

const ABILITY_NAMES: Record<string, string> = {
  strength: 'STR',
  dexterity: 'DEX',
  constitution: 'CON',
  intelligence: 'INT',
  wisdom: 'WIS',
  charisma: 'CHA',
}

export function AbilityScores({ scores }: AbilityScoresProps) {
  return (
    <div className="grid grid-cols-3 gap-2">
      {Object.entries(scores).map(([ability, value]) => {
        const modifier = Math.floor((value - 10) / 2)
        const modifierStr = modifier >= 0 ? `+${modifier}` : `${modifier}`

        return (
          <div
            key={ability}
            className="text-center p-2 bg-gray-100 rounded border"
          >
            <div className="text-xs text-gray-500 font-bold">
              {ABILITY_NAMES[ability]}
            </div>
            <div className="text-lg font-bold">{value}</div>
            <div className="text-sm text-primary-500">({modifierStr})</div>
          </div>
        )
      })}
    </div>
  )
}

interface SkillsListProps {
  skills: Record<string, number>
}

const SKILL_NAMES: Record<string, string> = {
  acrobatics: '杂技',
  animalHandling: '驯兽',
  arcana: '奥秘',
  athletics: '运动',
  deception: '欺瞒',
  history: '历史',
  insight: '洞察',
  intimidation: '威吓',
  investigation: '调查',
  medicine: '医药',
  nature: '自然',
  perception: '察觉',
  performance: '表演',
  persuasion: '说服',
  religion: '宗教',
  sleightOfHand: '巧手',
  stealth: '隐匿',
  survival: '生存',
}

export function SkillsList({ skills }: SkillsListProps) {
  const [expanded, setExpanded] = useState(false)

  const sortedSkills = Object.entries(skills)
    .sort((a, b) => b[1] - a[1])

  const displayedSkills = expanded ? sortedSkills : sortedSkills.slice(0, 6)

  return (
    <Panel title="技能">
      <div className="space-y-1">
        {displayedSkills.map(([skill, bonus]) => {
          const bonusStr = bonus >= 0 ? `+${bonus}` : `${bonus}`
          const isProficient = bonus > 0 // 简化判断

          return (
            <div
              key={skill}
              className={`
                flex justify-between text-sm py-1
                ${isProficient ? 'font-bold' : ''}
              `}
            >
              <span>{SKILL_NAMES[skill] || skill}</span>
              <span className={bonus >= 0 ? 'text-green-600' : 'text-red-600'}>
                {bonusStr}
              </span>
            </div>
          )
        })}
      </div>
      {sortedSkills.length > 6 && (
        <button
          onClick={() => setExpanded(!expanded)}
          className="w-full mt-2 text-sm text-primary-500 hover:underline"
        >
          {expanded ? '收起' : `显示全部 (${sortedSkills.length})`}
        </button>
      )}
    </Panel>
  )
}

// src/components/panels/InventoryPanel.tsx

import { useState } from 'react'
import { useGameStore } from '../../stores/gameStore'
import { Panel } from '../ui/Panel'
import { ItemDetailDialog } from '../dialogs/ItemDetailDialog'

export function InventoryPanel() {
  const [selectedItem, setSelectedItem] = useState<string | null>(null)
  const party = useGameStore((s) => s.gameState?.party)

```tsx
// src/components/panels/SpellbookPanel.tsx

import { useState } from 'react'
import { useGameStore } from '../../stores/gameStore'
import { Panel } from '../ui/Panel'
import { SpellDetailDialog } from '../dialogs/SpellDetailDialog'

export function SpellbookPanel() {
  const [selectedSpell, setSelectedSpell] = useState<string | null>(null)
  const party = useGameStore((s) => s.gameState?.party)
  const selectedCharId = useUIStore((s) => s.selectedCharacterId)

  const character = party?.find((c) => c.id === selectedCharId) || party?.[0]

  if (!character?.spellcasting) {
    return (
      <Panel title="法术">
        <div className="text-center text-gray-400 py-4">
          此角色无法施法
        </div>
      </Panel>
    )
  }

  const { spellcasting } = character

  // 按等级分组
  const spellsByLevel: Record<number, string[]> = {}
  spellcasting.knownSpells.forEach((spellId) => {
    const level = getSpellLevel(spellId) // 从数据获取法术等级
    if (!spellsByLevel[level]) {
      spellsByLevel[level] = []
    }
    spellsByLevel[level].push(spellId)
  })

  return (
    <>
      <Panel title="法术">
        {/* 法术位显示 */}
        <div className="mb-3 grid grid-cols-3 gap-1 text-center">
          {Object.entries(spellcasting.spellSlots).map(([level, remaining]) => {
            const max = getMaxSpellSlots(character.level, parseInt(level))
            return (
              <div key={level} className="text-xs">
                <div className="text-gray-500">{level}环</div>
                <div className="font-bold">{remaining}/{max}</div>
              </div>
            )
          })}
        </div>

        {/* 专注法术 */}
        {spellcasting.concentration && (
          <div className="mb-2 p-2 bg-blue-100 rounded text-sm">
            专注: {spellcasting.concentration}
          </div>
        )}

        {/* 法术列表 */}
        <div className="space-y-2 max-h-60 overflow-y-auto">
          {Object.entries(spellsByLevel).map(([level, spells]) => (
            <div key={level}>
              <div className="text-xs text-gray-500 mb-1">
                {level === '0' ? '戏法' : `${level}环法术`}
              </div>
              {spells.map((spellId) => (
                <div
                  key={spellId}
                  onClick={() => setSelectedSpell(spellId)}
                  className="p-2 bg-white rounded border hover:bg-gray-50 cursor-pointer text-sm"
                >
                  {spellId}
                </div>
              ))}
            </div>
          ))}
        </div>

        {/* 施法按钮 */}
        <button className="w-full mt-2 px-4 py-2 bg-primary-500 text-white rounded hover:bg-primary-600">
          施法
        </button>
      </Panel>

      {/* 法术详情弹窗 */}
      {selectedSpell && (
        <SpellDetailDialog
          spellId={selectedSpell}
          onClose={() => setSelectedSpell(null)}
        />
      )}
    </>
  )
}
```
