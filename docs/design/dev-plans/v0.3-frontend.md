#### 2. 游戏管理器

```typescript
// src/game/GameManager.ts

import Phaser from 'phaser'
import { gameConfig } from './config'
import { CombatScene } from './scenes/CombatScene'
import { SceneMapScene } from './scenes/SceneMapScene'
import { WorldMapScene } from './scenes/WorldMapScene'

export type SceneType = 'combat' | 'sceneMap' | 'worldMap'

export class GameManager {
  private game: Phaser.Game | null = null
  private currentScene: SceneType | null = null

  init(parent: string) {
    if (this.game) {
      this.game.destroy(true)
    }

    const config: Phaser.Types.Core.GameConfig = {
      ...gameConfig,
      parent,
      scene: [CombatScene, SceneMapScene, WorldMapScene],
    }

    this.game = new Phaser.Game(config)
  }

  switchScene(sceneType: SceneType, data?: any) {
    if (!this.game) return

    this.currentScene = sceneType
    this.game.scene.start(sceneType, data)
  }

  getCurrentScene(): Phaser.Scene | null {
    if (!this.game || !this.currentScene) return null
    return this.game.scene.getScene(this.currentScene)
  }

  destroy() {
    if (this.game) {
      this.game.destroy(true)
      this.game = null
    }
  }
}

// 单例
export const gameManager = new GameManager()
```

#### 3. 游戏容器组件

```tsx
// src/components/game/GameContainer.tsx

import { useEffect, useRef } from 'react'
import { gameManager } from '../../game/GameManager'
import { useGameStore } from '../../stores/gameStore'

interface GameContainerProps {
  className?: string
}

export function GameContainer({ className }: GameContainerProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  const phase = useGameStore((s) => s.gameState?.phase)

  useEffect(() => {
    if (containerRef.current) {
      gameManager.init(containerRef.current.id)
    }

    return () => {
      gameManager.destroy()
    }
  }, [])

  useEffect(() => {
    if (!phase) return

    switch (phase) {
      case 'combat':
        gameManager.switchScene('combat')
        break
      case 'exploring':
        gameManager.switchScene('sceneMap')
        break
      default:
        break
    }
  }, [phase])

  return (
    <div
      ref={containerRef}
      id="game-container"
      className={`w-full h-full ${className || ''}`}
    />
  )
}
```

### 实现指导

#### 1. 战斗场景

```typescript
// src/game/scenes/CombatScene.ts

import Phaser from 'phaser'
import { useGameStore } from '../../stores/gameStore'

const TILE_SIZE = 64
const GRID_COLOR = 0x444466
const MOVE_RANGE_COLOR = 0x44ff44
const ATTACK_RANGE_COLOR = 0xff4444

interface CombatSceneData {
  mapId: string
}

export class CombatScene extends Phaser.Scene {
  private gridGraphics!: Phaser.GameObjects.Graphics
  private unitSprites: Map<string, Phaser.GameObjects.Sprite> = new Map()
  private rangeGraphics!: Phaser.GameObjects.Graphics
  private currentUnit: string | null = null

  constructor() {
    super({ key: 'combat' })
  }

  init(data: CombatSceneData) {
    // 初始化战斗数据
  }

  create() {
    // 绘制网格
    this.drawGrid()

    // 创建单位
    this.createUnits()

    // 设置交互
    this.setupInteraction()

    // 监听状态更新
    this.listenForUpdates()
  }

  private drawGrid() {
    this.gridGraphics = this.add.graphics()
    this.gridGraphics.lineStyle(1, GRID_COLOR, 0.5)

    const width = this.cameras.main.width
    const height = this.cameras.main.height

    for (let x = 0; x < width; x += TILE_SIZE) {
      this.gridGraphics.moveTo(x, 0)
      this.gridGraphics.lineTo(x, height)
    }

    for (let y = 0; y < height; y += TILE_SIZE) {
      this.gridGraphics.moveTo(0, y)
      this.gridGraphics.lineTo(width, y)
    }

    this.gridGraphics.strokePath()
  }

  private createUnits() {
    const gameState = useGameStore.getState().gameState
    if (!gameState?.combat) return

    // 创建玩家单位
    gameState.party.forEach((char) => {
      const sprite = this.createUnitSprite(char.id, char.name, 0x4488ff)
      this.unitSprites.set(char.id, sprite)
    })

    // 创建敌人单位
    gameState.combat.participants.forEach((id) => {
      if (!this.unitSprites.has(id)) {
        const sprite = this.createUnitSprite(id, 'Enemy', 0xff4444)
        this.unitSprites.set(id, sprite)
      }
    })
  }

  private createUnitSprite(id: string, name: string, color: number): Phaser.GameObjects.Sprite {
    // 创建一个简单的圆形精灵
    const graphics = this.add.graphics()
    graphics.fillStyle(color, 1)
    graphics.fillCircle(TILE_SIZE / 2, TILE_SIZE / 2, TILE_SIZE / 2 - 4)

    // 生成纹理
    const key = `unit-${id}`
    graphics.generateTexture(key, TILE_SIZE, TILE_SIZE)
    graphics.destroy()

    // 创建精灵
    const sprite = this.add.sprite(0, 0, key)
    sprite.setData('unitId', id)

    // 添加名称文本
    const text = this.add.text(0, 0, name, {
      fontSize: '10px',
      color: '#ffffff',
    })
    text.setOrigin(0.5)
    text.setPosition(TILE_SIZE / 2, -10)

    // 创建容器
    const container = this.add.container(0, 0)
    container.add([sprite, text])
    container.setSize(TILE_SIZE, TILE_SIZE)

    return sprite
  }

  private setupInteraction() {
    this.input.on('gameobjectdown', (pointer: Phaser.Input.Pointer, gameObject: Phaser.GameObjects.GameObject) => {
      const unitId = gameObject.getData('unitId')
      if (unitId) {
        this.onUnitClick(unitId)
      }
    })

    this.input.on('pointerdown', (pointer: Phaser.Input.Pointer) => {
      const tileX = Math.floor(pointer.x / TILE_SIZE)
      const tileY = Math.floor(pointer.y / TILE_SIZE)
      this.onTileClick(tileX, tileY)
    })
  }

  private onUnitClick(unitId: string) {
    // 发送攻击请求
    const gameState = useGameStore.getState().gameState
    if (gameState?.combat && this.currentUnit) {
      // 通过 WebSocket 发送攻击
      const { send } = useWebSocket()
      send({
        type: 'combat_action',
        payload: {
          actionType: 'attack',
          sourceId: this.currentUnit,
          targetId: unitId,
        },
      })
    }
  }

  private onTileClick(tileX: number, tileY: number) {
    // 发送移动请求
    if (this.currentUnit) {
      const { send } = useWebSocket()
      send({
        type: 'combat_action',
        payload: {
          actionType: 'move',
          characterId: this.currentUnit,
          position: { x: tileX, y: tileY },
        },
      })
    }
  }

  showMoveRange(unitId: string, speed: number) {
    const sprite = this.unitSprites.get(unitId)
    if (!sprite) return

    const pos = this.getTilePosition(sprite)
    this.rangeGraphics.clear()
    this.rangeGraphics.fillStyle(MOVE_RANGE_COLOR, 0.3)

    // 简单的曼哈顿距离范围
    for (let dx = -speed; dx <= speed; dx++) {
      for (let dy = -speed; dy <= speed; dy++) {
        if (Math.abs(dx) + Math.abs(dy) <= speed) {
          this.rangeGraphics.fillRect(
            (pos.x + dx) * TILE_SIZE,
            (pos.y + dy) * TILE_SIZE,
            TILE_SIZE,
            TILE_SIZE
          )
        }
      }
    }
  }

  showAttackRange(unitId: string, range: number) {
    const sprite = this.unitSprites.get(unitId)
    if (!sprite) return

    const pos = this.getTilePosition(sprite)
    this.rangeGraphics.clear()
    this.rangeGraphics.fillStyle(ATTACK_RANGE_COLOR, 0.3)

    for (let dx = -range; dx <= range; dx++) {
      for (let dy = -range; dy <= range; dy++) {
        if (Math.abs(dx) + Math.abs(dy) <= range) {
          this.rangeGraphics.fillRect(
            (pos.x + dx) * TILE_SIZE,
            (pos.y + dy) * TILE_SIZE,
            TILE_SIZE,
            TILE_SIZE
          )
        }
      }
    }
  }

  clearRange() {
    this.rangeGraphics.clear()
  }

  private getTilePosition(sprite: Phaser.GameObjects.Sprite): { x: number; y: number } {
    return {
      x: Math.floor(sprite.x / TILE_SIZE),
      y: Math.floor(sprite.y / TILE_SIZE),
    }
  }

  private listenForUpdates() {
    // 监听 WebSocket 消息更新单位位置
    const { subscribe } = useWebSocket()

    subscribe((message) => {
      if (message.type === 'combat_event') {
        const event = message.payload as any
        this.handleCombatEvent(event)
      }
    })
  }

  private handleCombatEvent(event: any) {
    switch (event.type) {
      case 'move':
        this.animateMove(event.characterId, event.from, event.to)
        break
      case 'attack':
        this.animateAttack(event.sourceId, event.targetId)
        break
      case 'damage':
        this.showDamageNumber(event.targetId, event.damage)
        break
    }
  }

  private animateMove(unitId: string, from: { x: number; y: number }, to: { x: number; y: number }) {
    const sprite = this.unitSprites.get(unitId)
    if (!sprite) return

    this.tweens.add({
      targets: sprite,
      x: to.x * TILE_SIZE + TILE_SIZE / 2,
      y: to.y * TILE_SIZE + TILE_SIZE / 2,
      duration: 300,
      ease: 'Power2',
    })
  }

  private animateAttack(sourceId: string, targetId: string) {
    const sourceSprite = this.unitSprites.get(sourceId)
    const targetSprite = this.unitSprites.get(targetId)

    if (!sourceSprite || !targetSprite) return

    // 简单的攻击动画：向目标移动然后返回
    const originalX = sourceSprite.x
    const originalY = sourceSprite.y

    this.tweens.add({
      targets: sourceSprite,
      x: targetSprite.x,
      y: targetSprite.y,
      duration: 150,
      yoyo: true,
      ease: 'Power2',
    })
  }

  private showDamageNumber(targetId: string, damage: number) {
    const sprite = this.unitSprites.get(targetId)
    if (!sprite) return

    const text = this.add.text(sprite.x, sprite.y - 20, `-${damage}`, {
      fontSize: '24px',
      color: '#ff4444',
      fontStyle: 'bold',
    })
    text.setOrigin(0.5)

    this.tweens.add({
      targets: text,
      y: text.y - 40,
      alpha: 0,
      duration: 1000,
      onComplete: () => text.destroy(),
    })
  }
}
```
