#### 1. 骰子表达式解析

```go
// internal/server/dice/roller.go
package dice

import (
    "math/rand"
    "regexp"
    "strconv"
    "strings"
)

// Roller 骰子掷骰器
type Roller struct {
    dice     []DieSpec
    modifier int
    keep     int // 保留数量，0 表示全部
}

type DieSpec struct {
    Count int
    Sides int
}

// Parse 解析骰子表达式
// 支持: d20, 2d6, 4d6k3, 2d6+3, 1d20-2
func Parse(formula string) (*Roller, error) {
    // 正则解析: ([\d]*)d(\d+)(k(\d+))?([+-]\d+)?
    re := regexp.MustCompile(`^(\d*)d(\d+)(k(\d+))?([+-]\d+)?$`)
    matches := re.FindStringSubmatch(strings.ToLower(formula))

    if matches == nil {
        return nil, fmt.Errorf("invalid dice formula: %s", formula)
    }

    roller := &Roller{}

    // 解析骰子数量
    if matches[1] == "" {
        roller.dice = []DieSpec{{Count: 1, Sides: mustAtoi(matches[2])}}
    } else {
        count := mustAtoi(matches[1])
        sides := mustAtoi(matches[2])
        roller.dice = []DieSpec{{Count: count, Sides: sides}}
    }

    // 解析保留数量
    if matches[4] != "" {
        roller.keep = mustAtoi(matches[4])
    }

    // 解析修饰符
    if matches[5] != "" {
        roller.modifier = mustAtoi(matches[5])
    }

    return roller, nil
}

// Roll 执行掷骰
func (r *Roller) Roll(rnd *rand.Rand) *models.DiceResult {
    result := &models.DiceResult{
        Modifier: r.modifier,
    }

    var allRolls []int
    for _, die := range r.dice {
        for i := 0; i < die.Count; i++ {
            roll := rnd.Intn(die.Sides) + 1
            allRolls = append(allRolls, roll)
        }
    }

    // 保留最高的 N 个
    if r.keep > 0 && r.keep < len(allRolls) {
        sort.Sort(sort.Reverse(sort.IntSlice(allRolls)))
        result.Dice = allRolls[:r.keep]
    } else {
        result.Dice = allRolls
    }

    // 计算总和
    sum := 0
    for _, d := range result.Dice {
        sum += d
    }
    result.Total = sum + r.modifier

    // 检查暴击/大失败（仅 d20）
    if len(result.Dice) == 1 {
        if result.Dice[0] == 20 {
            result.IsCrit = true
        } else if result.Dice[0] == 1 {
            result.IsFumble = true
        }
    }

    return result
}
```

#### 2. 属性检定

```go
// internal/server/dice/check.go
package dice

// CheckResult 检定结果
type CheckResult struct {
    Success   bool   `json:"success"`
    Roll      int    `json:"roll"`
    Modifier  int    `json:"modifier"`
    Total     int    `json:"total"`
    DC        int    `json:"dc"`
    Advantage bool   `json:"advantage"`
    Disadvantage bool `json:"disadvantage"`
}

// AbilityCheck 属性检定
func AbilityCheck(modifier int, dc int, advantage, disadvantage bool) *CheckResult {
    result := &CheckResult{
        Modifier: modifier,
        DC: dc,
        Advantage: advantage,
        Disadvantage: disadvantage,
    }

    if advantage && !disadvantage {
        // 优势：取两次中较高的
        r1 := rollD20()
        r2 := rollD20()
        result.Roll = max(r1, r2)
    } else if disadvantage && !advantage {
        // 劣势：取两次中较低的
        r1 := rollD20()
        r2 := rollD20()
        result.Roll = min(r1, r2)
    } else {
        result.Roll = rollD20()
    }

    result.Total = result.Roll + result.Modifier
    result.Success = result.Total >= result.DC

    return result
}

// GetModifier 根据属性值计算修正值
func GetModifier(score int) int {
    return (score - 10) / 2
}
```

```go
// internal/client/tools/registry.go
package tools

type ToolRegistry struct {
    tools map[string]Tool
}

type Tool struct {
    Name        string                 `json:"name"`
    Description string                 `json:"description"`
    InputSchema map[string]interface{} `json:"inputSchema"`
    Handler     ToolHandler            `json:"-"`
}

type ToolHandler func(args map[string]interface{}) (interface{}, error)

func NewRegistry() *ToolRegistry {
    return &ToolRegistry{
        tools: make(map[string]Tool),
    }
}

func (r *ToolRegistry) Register(tool Tool) {
    r.tools[tool.Name] = tool
}

func (r *ToolRegistry) Get(name string) (Tool, bool) {
    tool, ok := r.tools[name]
    return tool, ok
}

func (r *ToolRegistry) List() []Tool {
    result := make([]Tool, 0, len(r.tools))
    for _, tool := range r.tools {
        result = append(result, tool)
    }
    return result
}

func (r *ToolRegistry) Execute(name string, args map[string]interface{}) (interface{}, error) {
    tool, ok := r.Get(name)
    if !ok {
        return nil, fmt.Errorf("tool not found: %s", name)
    }
    return tool.Handler(args)
}
```

```go
// internal/client/llm/provider.go
package llm

import "context"

// Provider LLM 提供者接口
type Provider interface {
    // SendMessage 发送消息（非流式）
    SendMessage(ctx context.Context, req *Request) (*Response, error)

    // StreamMessage 发送消息（流式）
    StreamMessage(ctx context.Context, req *Request) (<-chan StreamChunk, error)

    // GetTools 获取工具定义
    GetTools() []ToolDefinition
}

// Request 请求结构
type Request struct {
    Messages []Message `json:"messages"`
    Tools    []ToolDefinition `json:"tools,omitempty"`
}

// Message 消息结构
type Message struct {
    Role    string `json:"role"` // system, user, assistant, tool
    Content string `json:"content"`
    ToolCalls []ToolCall `json:"toolCalls,omitempty"`
    ToolCallID string `json:"toolCallId,omitempty"`
}

// Response 响应结构
type Response struct {
    Content   string     `json:"content"`
    ToolCalls []ToolCall `json:"toolCalls,omitempty"`
    Usage     Usage      `json:"usage"`
}

// ToolCall 工具调用
type ToolCall struct {
    ID       string                 `json:"id"`
    Name     string                 `json:"name"`
    Arguments map[string]interface{} `json:"arguments"`
}

// StreamChunk 流式响应块
type StreamChunk struct {
    Delta     string     `json:"delta,omitempty"`
    ToolCall  *ToolCall  `json:"toolCall,omitempty"`
    Done      bool       `json:"done"`
    Error     error      `json:"error,omitempty"`
}

// ToolDefinition 工具定义
type ToolDefinition struct {
    Type        string                 `json:"type"`
    Function    FunctionDefinition     `json:"function"`
}

type FunctionDefinition struct {
    Name        string                 `json:"name"`
    Description string                 `json:"description"`
    Parameters  map[string]interface{} `json:"parameters"`
}
```

#### 2. OpenAI 适配器

```go
// internal/client/llm/openai.go
package llm

import (
    "context"
    "github.com/sashabaranov/go-openai"
)

type OpenAIProvider struct {
    client *openai.Client
    config *OpenAIConfig
}

type OpenAIConfig struct {
    APIKey  string
    Model   string
    BaseURL string
}

func NewOpenAIProvider(config *OpenAIConfig) *OpenAIProvider {
    clientConfig := openai.DefaultConfig(config.APIKey)
    if config.BaseURL != "" {
        clientConfig.BaseURL = config.BaseURL
    }

    return &OpenAIProvider{
        client: openai.NewClientWithConfig(clientConfig),
        config: config,
    }
}

func (p *OpenAIProvider) StreamMessage(ctx context.Context, req *Request) (<-chan StreamChunk, error) {
    stream := make(chan StreamChunk, 100)

    go func() {
        defer close(stream)

        // 转换消息格式
        messages := p.convertMessages(req.Messages)

        // 创建流式请求
        streamReq := openai.ChatCompletionRequest{
            Model:    p.config.Model,
            Messages: messages,
            Stream:   true,
        }

        // 添加工具定义
        if len(req.Tools) > 0 {
            streamReq.Tools = p.convertTools(req.Tools)
        }

        resp, err := p.client.CreateChatCompletionStream(ctx, streamReq)
        if err != nil {
            stream <- StreamChunk{Error: err}
            return
        }
        defer resp.Close()

        for {
            chunk, err := resp.Recv()
            if err != nil {
                if err == io.EOF {
                    stream <- StreamChunk{Done: true}
                    return
                }
                stream <- StreamChunk{Error: err}
                return
            }

            if len(chunk.Choices) > 0 {
                delta := chunk.Choices[0].Delta

                // 文本内容
                if delta.Content != "" {
                    stream <- StreamChunk{Delta: delta.Content}
                }

                // 工具调用
                if len(delta.ToolCalls) > 0 {
                    tc := delta.ToolCalls[0]
                    stream <- StreamChunk{
                        ToolCall: &ToolCall{
                            ID:   tc.ID,
                            Name: tc.Function.Name,
                            Arguments: parseJSON(tc.Function.Arguments),
                        },
                    }
                }
            }
        }
    }()

    return stream, nil
}
```

```go
// internal/api/websocket/hub.go
package websocket

import (
    "github.com/gorilla/websocket"
)

type Hub struct {
    clients      map[*Client]bool
    broadcast    chan []byte
    register     chan *Client
    unregister   chan *Client
    stateManager *state.Manager
    session      *session.Manager
}

func NewHub(stateManager *state.Manager) *Hub {
    return &Hub{
        clients:      make(map[*Client]bool),
        broadcast:    make(chan []byte, 256),
        register:     make(chan *Client),
        unregister:   make(chan *Client),
        stateManager: stateManager,
    }
}

func (h *Hub) Run() {
    for {
        select {
        case client := <-h.register:
            h.clients[client] = true

        case client := <-h.unregister:
            if _, ok := h.clients[client]; ok {
                delete(h.clients, client)
                close(client.send)
            }

        case message := <-h.broadcast:
            for client := range h.clients {
                select {
                case client.send <- message:
                default:
                    close(client.send)
                    delete(h.clients, client)
                }
            }
        }
    }
}
```

#### 2. 消息处理

```go
// internal/api/websocket/handler.go
package websocket

func (h *Hub) HandleMessage(client *Client, data []byte) {
    var msg models.ClientMessage
    if err := json.Unmarshal(data, &msg); err != nil {
        client.SendError("invalid message format")
        return
    }

    switch msg.Type {
    case models.MsgTypeUserInput:
        h.handleUserInput(client, msg)

    case models.MsgTypeManagement:
        h.handleManagement(client, msg)

    case models.MsgTypePing:
        client.SendPong()

    default:
        client.SendError("unknown message type")
    }
}

func (h *Hub) handleUserInput(client *Client, msg models.ClientMessage) {
    var payload struct {
        Text string `json:"text"`
    }
    if err := json.Unmarshal(msg.Payload, &payload); err != nil {
        client.SendError("invalid payload")
        return
    }

    // 发送到 LLM 会话
    go func() {
        stream, err := h.session.SendMessage(client.ctx, payload.Text)
        if err != nil {
            client.SendError(err.Error())
            return
        }

        // 流式推送叙述
        var fullText strings.Builder
        for chunk := range stream {
            if chunk.Error != nil {
                client.SendError(chunk.Error.Error())
                return
            }
            if chunk.Done {
                // 发送完成信号
                client.SendMessage(models.ServerMessage{
                    Type: models.MsgTypeNarration,
                    Payload: map[string]interface{}{
                        "text":       fullText.String(),
                        "isStreaming": false,
                    },
                    RequestID: msg.RequestID,
                })
                return
            }
            if chunk.Delta != "" {
                fullText.WriteString(chunk.Delta)
                client.SendMessage(models.ServerMessage{
                    Type: models.MsgTypeNarration,
                    Payload: map[string]interface{}{
                        "text":       chunk.Delta,
                        "isStreaming": true,
                    },
                })
            }
            if chunk.ToolCall != nil {
                // 执行工具调用
                result, err := h.toolRegistry.Execute(
                    chunk.ToolCall.Name,
                    chunk.ToolCall.Arguments,
                )
                if err != nil {
                    h.session.AddToolResult(chunk.ToolCall.ID, err.Error())
                } else {
                    h.session.AddToolResult(chunk.ToolCall.ID, result)

                    // 推送骰子结果
                    if chunk.ToolCall.Name == "roll_dice" {
                        client.SendMessage(models.ServerMessage{
                            Type: models.MsgTypeDiceResult,
                            Payload: result,
                        })
                    }
                }
            }
        }
    }()
}
```

#### 1. 持久化管理器

```go
// internal/persistence/manager.go
package persistence

import (
    "encoding/json"
    "os"
    "path/filepath"
)

type Manager struct {
    basePath string
}

func NewManager(basePath string) *Manager {
    return &Manager{basePath: basePath}
}

// CreateSession 创建会话目录
func (m *Manager) CreateSession(sessionID string) error {
    dir := filepath.Join(m.basePath, "sessions", sessionID)
    return os.MkdirAll(dir, 0755)
}

// AppendMessage 追加消息到 JSONL
func (m *Manager) AppendMessage(sessionID string, msg *models.Message) error {
    path := filepath.Join(m.basePath, "sessions", sessionID, "messages.jsonl")
    f, err := os.OpenFile(path, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0644)
    if err != nil {
        return err
    }
    defer f.Close()

    data, err := json.Marshal(msg)
    if err != nil {
        return err
    }

    _, err = f.Write(append(data, '\n'))
    return err
}

// LoadMessages 加载消息历史
func (m *Manager) LoadMessages(sessionID string, offset, limit int) ([]*models.Message, error) {
    path := filepath.Join(m.basePath, "sessions", sessionID, "messages.jsonl")
    f, err := os.Open(path)
    if err != nil {
        return nil, err
    }
    defer f.Close()

    var messages []*models.Message
    scanner := bufio.NewScanner(f)
    lineNum := 0

    for scanner.Scan() {
        if lineNum < offset {
            lineNum++
            continue
        }
        if limit > 0 && len(messages) >= limit {
            break
        }

        var msg models.Message
        if err := json.Unmarshal(scanner.Bytes(), &msg); err != nil {
            continue
        }
        messages = append(messages, &msg)
        lineNum++
    }

    return messages, scanner.Err()
}

// SaveState 保存状态快照
func (m *Manager) SaveState(sessionID string, state *state.GameState) error {
    path := filepath.Join(m.basePath, "sessions", sessionID, "state.json")
    f, err := os.Create(path)
    if err != nil {
        return err
    }
    defer f.Close()

    encoder := json.NewEncoder(f)
    encoder.SetIndent("", "  ")
    return encoder.Encode(state)
}

// LoadState 加载状态
func (m *Manager) LoadState(sessionID string) (*state.GameState, error) {
    path := filepath.Join(m.basePath, "sessions", sessionID, "state.json")
    f, err := os.Open(path)
    if err != nil {
        return nil, err
    }
    defer f.Close()

    var gameState state.GameState
    decoder := json.NewDecoder(f)
    if err := decoder.Decode(&gameState); err != nil {
        return nil, err
    }

    return &gameState, nil
}
```
