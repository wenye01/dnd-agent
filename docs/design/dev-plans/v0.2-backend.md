#### 1. 基础角色模型

```go
// internal/shared/models/character.go
package models

type Character struct {
    ID               string        `json:"id"`
    Name             string        `json:"name"`
    Race             string        `json:"race"`
    Class            string        `json:"class"`
    Level            int           `json:"level"`
    Background       string        `json:"background"`
    AbilityScores    AbilityScores `json:"abilityScores"`
    MaxHitPoints     int           `json:"maxHitPoints"`
    CurrentHitPoints int           `json:"currentHitPoints"`
    ArmorClass       int           `json:"armorClass"`
    Speed            int           `json:"speed"`
    ProficiencyBonus int           `json:"proficiencyBonus"`
    Skills           map[string]int `json:"skills"`
    SavingThrows     []string       `json:"savingThrows"`
    Inventory        []*InventoryItem `json:"inventory"`
    Gold             int            `json:"gold"`
}

// AbilityScores 使用 string key 方便 JSON
type AbilityScores map[string]int // "str", "dex", "con", "int", "wis", "cha"

// AbilityModifier 计算属性修正符
func AbilityModifier(score int) int {
    return (score - 10) / 2
}

// ProficiencyBonus 按等级计算熟练加值
func ProficiencyBonusForLevel(level int) int {
    return (level-1)/4 + 2 // 1-4: +2, 5-8: +3, ...
}
```

#### 2. 基础角色创建（原型版）

```go
// internal/server/character/creator.go
package character

// CreateParams 角色创建参数
type CreateParams struct {
    Name          string
    Race          string
    Class         string
    Background    string
    AbilityScores map[string]int
}

// CreateBasic 创建基础角色（原型版，仅支持有限种族/职业）
func CreateBasic(params CreateParams) (*models.Character, error) {
    char := &models.Character{
        ID:    uuid.New().String(),
        Name:  params.Name,
        Race:  params.Race,
        Class: params.Class,
        Level: 1,
        Background:    params.Background,
        AbilityScores: params.AbilityScores,
    }

    // 计算熟练加值
    char.ProficiencyBonus = models.ProficiencyBonusForLevel(1)

    // 按职业设置命中骰和基础 HP
    conMod := models.AbilityModifier(params.AbilityScores["con"])
    switch params.Class {
    case "fighter":
        char.MaxHitPoints = 10 + conMod
    case "wizard":
        char.MaxHitPoints = 6 + conMod
    case "rogue":
        char.MaxHitPoints = 8 + conMod
    default:
        char.MaxHitPoints = 8 + conMod
    }
    char.CurrentHitPoints = char.MaxHitPoints

    // 基础 AC（无护甲）
    dexMod := models.AbilityModifier(params.AbilityScores["dex"])
    char.ArmorClass = 10 + dexMod
    char.Speed = 30

    return char, nil
}
```

#### 1. 工具注册表

```go
// internal/client/tools/registry.go
package tools

import "encoding/json"

// ToolDefinition 工具定义
type ToolDefinition struct {
    Name        string                 `json:"name"`
    Description string                 `json:"description"`
    InputSchema map[string]interface{} `json:"input_schema"` // JSON Schema
    Handler     func(args map[string]interface{}) (interface{}, error)
}

// Registry 工具注册表
type Registry struct {
    tools map[string]*ToolDefinition
}

func NewRegistry() *Registry {
    return &Registry{tools: make(map[string]*ToolDefinition)}
}

func (r *Registry) Register(def *ToolDefinition) {
    r.tools[def.Name] = def
}

// Execute 执行工具，返回 JSON 结果字符串
func (r *Registry) Execute(name string, args map[string]interface{}) (string, error) {
    tool, ok := r.tools[name]
    if !ok {
        return "", fmt.Errorf("unknown tool: %s", name)
    }
    result, err := tool.Handler(args)
    if err != nil {
        return fmt.Sprintf(`{"error": "%s"}`, err.Error()), nil // 错误返回给 LLM
    }
    data, _ := json.Marshal(result)
    return string(data), nil
}

// GetSchemas 返回所有工具的 Schema（发送给 LLM）
func (r *Registry) GetSchemas() []map[string]interface{} {
    schemas := make([]map[string]interface{}, 0, len(r.tools))
    for _, tool := range r.tools {
        schemas = append(schemas, map[string]interface{}{
            "name":         tool.Name,
            "description":  tool.Description,
            "input_schema": tool.InputSchema,
        })
    }
    return schemas
}
```

```go
type Session struct {

    llm       llm.Provider

func NewSession(id string, llm llm.Provider, tools *tools.Registry, gameState *state.GameState) *Session {
    return &Session{
        ID:        id,
        Messages:  []llm.Message{},
        GameState: gameState,
        llm:       llm,
        tools:     tools,
    }
}
```

#### 2. 主处理循环（工具调用 agentic loop）

```go
// SendMessage 发送用户消息，执行 agentic loop 直到 LLM 停止调用工具
func (s *Session) SendMessage(ctx context.Context, userText string, streamCh chan<- string) error {
    // 1. 追加用户消息
    s.Messages = append(s.Messages, llm.Message{Role: "user", Content: userText})

    for {
        // 2. 构建 system prompt
        systemPrompt := s.buildSystemPrompt()

        // 3. 调用 LLM（流式）
        response, err := s.llm.StreamMessage(ctx, systemPrompt, s.Messages, s.tools.GetSchemas(), streamCh)
        if err != nil {
            return err
        }

        // 4. 追加 LLM 响应
        s.Messages = append(s.Messages, llm.Message{Role: "assistant", Content: response.Content, ToolCalls: response.ToolCalls})

        // 5. 没有工具调用则结束
        if len(response.ToolCalls) == 0 {
            break
        }

        // 6. 执行工具调用，追加结果
        for _, call := range response.ToolCalls {
            result, _ := s.tools.Execute(call.Name, call.Arguments)
            s.Messages = append(s.Messages, llm.Message{
                Role:       "tool",
                ToolCallID: call.ID,
                Content:    result,
            })
        }
    }
    return nil
}

func (s *Session) buildSystemPrompt() string {
    // v0.1 基础版：包含角色信息和游戏状态摘要
    return fmt.Sprintf(`你是一位 D&D 地下城主（DM）。
当前游戏状态：%s

规则：严格按照 D&D 5e 规则执行游戏。通过工具调用执行游戏操作。`, s.GameState.Summary())
}
```

#### 3. 会话管理器

```go
// internal/client/session/manager.go
type Manager struct {
    sessions map[string]*Session
    mu       sync.RWMutex
}

func (m *Manager) Create(sessionID string, llm llm.Provider, tools *tools.Registry, gameState *state.GameState) *Session {
    m.mu.Lock()
    defer m.mu.Unlock()
    sess := NewSession(sessionID, llm, tools, gameState)
    m.sessions[sessionID] = sess
    return sess
}

func (m *Manager) Get(sessionID string) (*Session, bool) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    sess, ok := m.sessions[sessionID]
    return sess, ok
}
```

```go
// internal/api/rest/router.go
package rest

func SetupRouter(sessionMgr *session.Manager, gameState *state.Manager) *gin.Engine {
    r := gin.New()
    r.Use(gin.Recovery(), LoggerMiddleware())

    api := r.Group("/api")
    {
        // 会话管理
        api.POST("/sessions", createSession(sessionMgr, gameState))
        api.GET("/sessions/:id", getSession(sessionMgr))
        api.DELETE("/sessions/:id", deleteSession(sessionMgr))

        // 配置
        api.GET("/config", getConfig())

        // 健康检查
        api.GET("/health", func(c *gin.Context) {
            c.JSON(200, gin.H{"status": "ok"})
        })
    }
    return r
}
```

#### 2. 统一响应格式

```go
// internal/api/rest/response.go
type APIResponse struct {
    Success bool        `json:"success"`
    Data    interface{} `json:"data,omitempty"`
    Error   string      `json:"error,omitempty"`
}

func OK(c *gin.Context, data interface{}) {
    c.JSON(200, APIResponse{Success: true, Data: data})
}

func Fail(c *gin.Context, status int, msg string) {
    c.JSON(status, APIResponse{Success: false, Error: msg})
}
```

#### 3. 创建会话接口

```go
func createSession(mgr *session.Manager, stateMgr *state.Manager) gin.HandlerFunc {
    return func(c *gin.Context) {
        var req struct {
            ScenarioID string `json:"scenarioId"`
        }
        if err := c.ShouldBindJSON(&req); err != nil {
            Fail(c, 400, "invalid request")
            return
        }
        sessionID := uuid.New().String()
        gameState := stateMgr.CreateNew(sessionID, req.ScenarioID)
        // session 将在 WebSocket 连接时关联
        OK(c, gin.H{"sessionId": sessionID, "state": gameState})
    }
}
```
