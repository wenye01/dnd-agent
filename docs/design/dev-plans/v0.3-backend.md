#### 2. 角色创建服务

```go
// internal/server/character/creator.go

package character

import (
    "dnd-game/backend/internal/shared/models"
    "dnd-game/backend/pkg/dnd5e"
)

type Creator struct {
    races    *dnd5e.RaceData
    classes  *dnd5e.ClassData
    backgrounds *dnd5e.BackgroundData
}

type CreateParams struct {
    Name        string
    Race        string
    Class       string
    Background  string
    Abilities   map[string]int  // 可选，不提供则自动生成
}

func (c *Creator) Create(params CreateParams) (*models.Character, error) {
    char := &models.Character{
        ID:          generateID(),
        Name:        params.Name,
        Race:        params.Race,
        Class:       params.Class,
        Background:  params.Background,
        Level:       1,
    }

    // 获取种族数据
    race, err := c.races.Get(params.Race)
    if err != nil {
        return nil, err
    }

    // 获取职业数据
    class, err := c.classes.Get(params.Class)
    if err != nil {
        return nil, err
    }

    // 生成或使用提供的属性
    if params.Abilities != nil {
        char.AbilityScores = params.Abilities
    } else {
        char.AbilityScores = c.generateAbilityScores(race)
    }

    // 应用种族加成
    for ability, bonus := range race.AbilityBonus {
        char.AbilityScores[ability] += bonus
    }

    // 计算基础属性
    char.ProficiencyBonus = 2
    char.MaxHitPoints = class.HitDie + GetModifier(char.AbilityScores["constitution"])
    char.CurrentHitPoints = char.MaxHitPoints
    char.HitDice = fmt.Sprintf("1d%d", class.HitDie)
    char.Speed = race.Speed

    // 计算技能
    char.Skills = c.calculateSkills(char, class, race)

    // 计算豁免
    char.SavingThrows = class.SavingThrows

    // 计算 AC（无装备）
    char.ArmorClass = 10 + GetModifier(char.AbilityScores["dexterity"])

    // 初始化装备槽
    char.Equipment = make(map[string]string)
    char.Inventory = make([]*models.InventoryItem, 0)

    // 添加起始装备
    for _, item := range class.StartingEquipment {
        char.Inventory = append(char.Inventory, &models.InventoryItem{
            ID:     generateID(),
            ItemID: item.ID,
            Count:  item.Count,
        })
    }

    // 初始化法术（如果职业有施法能力）
    if class.Spellcasting != nil {
        char.Spellcasting = &models.Spellcasting{
            Ability:     class.Spellcasting.Ability,
            SpellSlots:  c.getInitialSpellSlots(class),
            KnownSpells: class.Spellcasting.InitialSpells,
        }
        char.Spellcasting.SpellAttack = char.ProficiencyBonus +
            GetModifier(char.AbilityScores[class.Spellcasting.Ability])
        char.Spellcasting.SpellSaveDC = 8 + char.ProficiencyBonus +
            GetModifier(char.AbilityScores[class.Spellcasting.Ability])
    }

    return char, nil
}

func (c *Creator) generateAbilityScores(race *dnd5e.Race) map[string]int {
    // 使用 4d6k3 方法
    scores := make(map[string]int)
    abilities := []string{"strength", "dexterity", "constitution", "intelligence", "wisdom", "charisma"}

    for _, ability := range abilities {
        roller, _ := dice.Parse("4d6k3")
        result := roller.Roll(rand.New(rand.NewSource(time.Now().UnixNano())))
        scores[ability] = result.Total
    }

    return scores
}

func (c *Creator) calculateSkills(char *models.Character, class *dnd5e.Class, race *dnd5e.Race) map[string]int {
    skills := make(map[string]int)

    // 所有技能
    allSkills := []string{
        "acrobatics", "animalHandling", "arcana", "athletics",
        "deception", "history", "insight", "intimidation",
        "investigation", "medicine", "nature", "perception",
        "performance", "persuasion", "religion", "sleightOfHand",
        "stealth", "survival",
    }

    // 技能对应属性
    skillAbilities := map[string]string{
        "acrobatics":    "dexterity",
        "animalHandling": "wisdom",
        "arcana":        "intelligence",
        "athletics":     "strength",
        "deception":     "charisma",
        "history":       "intelligence",
        "insight":       "wisdom",
        "intimidation":  "charisma",
        "investigation": "intelligence",
        "medicine":      "wisdom",
        "nature":        "intelligence",
        "perception":    "wisdom",
        "performance":   "charisma",
        "persuasion":    "charisma",
        "religion":      "intelligence",
        "sleightOfHand": "dexterity",
        "stealth":       "dexterity",
        "survival":      "wisdom",
    }

    for _, skill := range allSkills {
        ability := skillAbilities[skill]
        mod := GetModifier(char.AbilityScores[ability])

        // 检查是否熟练
        isProficient := false
        for _, s := range class.SkillProficiencies {
            if s == skill {
                isProficient = true
                break
            }
        }

        if isProficient {
            mod += char.ProficiencyBonus
        }

        skills[skill] = mod
    }

    // 添加种族技能熟练
    for _, skill := range race.SkillProficiencies {
        skills[skill] += char.ProficiencyBonus
    }

    return skills
}
```

```go
// internal/server/combat/manager.go

package combat

import (
    "dnd-game/backend/internal/shared/models"
    "dnd-game/backend/internal/shared/state"
)

type Manager struct {
    stateManager *state.Manager
    dice         *dice.Service
}

type CombatState string

const (
    CombatStateIdle    CombatState = "idle"
    CombatStateActive  CombatState = "active"
    CombatStateEnded   CombatState = "ended"
)

// StartCombat 开始战斗
func (m *Manager) StartCombat(participantIDs []string) (*state.CombatState, error) {
    combat := &state.CombatState{
        Round:        1,
        TurnIndex:    0,
        Participants: participantIDs,
        Initiatives:  make([]*state.InitiativeEntry, 0),
        ActiveEffects: make([]*state.ActiveEffect, 0),
    }

    // 掷先攻
    for _, id := range participantIDs {
        char, err := m.stateManager.GetCharacter(id)
        if err != nil {
            return nil, err
        }

        // 掷 d20 + 敏捷修正
        roller, _ := dice.Parse("d20")
        result := roller.Roll(m.dice.Rand())
        initiative := result.Total + dice.GetModifier(char.AbilityScores["dexterity"])

        combat.Initiatives = append(combat.Initiatives, &state.InitiativeEntry{
            CharacterID: id,
            Initiative:  initiative,
            HasActed:    false,
        })
    }

    // 按先攻排序
    sort.Slice(combat.Initiatives, func(i, j int) bool {
        return combat.Initiatives[i].Initiative > combat.Initiatives[j].Initiative
    })

    // 更新游戏状态
    m.stateManager.UpdateGameState(func(gs *state.GameState) {
        gs.Phase = state.PhaseCombat
        gs.Combat = combat
    })

    return combat, nil
}

// NextTurn 下一回合
func (m *Manager) NextTurn() (*state.InitiativeEntry, error) {
    combat := m.stateManager.GetCombatState()
    if combat == nil {
        return nil, errors.New("no active combat")
    }

    // 标记当前角色已行动
    if combat.TurnIndex < len(combat.Initiatives) {
        combat.Initiatives[combat.TurnIndex].HasActed = true
    }

    // 移动到下一个角色
    combat.TurnIndex++

    // 检查是否需要新回合
    if combat.TurnIndex >= len(combat.Initiatives) {
        combat.Round++
        combat.TurnIndex = 0
        // 重置所有角色的行动状态
        for _, entry := range combat.Initiatives {
            entry.HasActed = false
        }
        // 处理回合开始的效果
        m.processStartOfTurnEffects(combat)
    }

    return combat.Initiatives[combat.TurnIndex], nil
}

// GetCurrentTurn 获取当前回合角色
func (m *Manager) GetCurrentTurn() (*state.InitiativeEntry, error) {
    combat := m.stateManager.GetCombatState()
    if combat == nil || combat.TurnIndex >= len(combat.Initiatives) {
        return nil, errors.New("no active turn")
    }
    return combat.Initiatives[combat.TurnIndex], nil
}
```

#### 2. 攻击处理

```go
// internal/server/combat/attack.go

package combat

type AttackParams struct {
    AttackerID string
    TargetID   string
    WeaponID   string // 可选，默认使用装备的武器
}

type AttackResult struct {
    AttackRoll   *dice.CheckResult `json:"attackRoll"`
    Hit          bool              `json:"hit"`
    Critical     bool              `json:"critical"`
    DamageRoll   *dice.DiceResult  `json:"damageRoll,omitempty"`
    DamageDealt  int               `json:"damageDealt,omitempty"`
    DamageType   string            `json:"damageType,omitempty"`
    TargetHP     int               `json:"targetHp"`
    TargetMaxHP  int               `json:"targetMaxHp"`
    TargetKilled bool              `json:"targetKilled"`
}

func (m *Manager) Attack(params AttackParams) (*AttackResult, error) {
    result := &AttackResult{}

    // 获取攻击者
    attacker, err := m.stateManager.GetCharacter(params.AttackerID)
    if err != nil {
        return nil, err
    }

    // 获取目标
    target, err := m.stateManager.GetCharacter(params.TargetID)
    if err != nil {
        return nil, err
    }

    // 获取武器
    weapon := m.getWeapon(attacker, params.WeaponID)

    // 计算攻击加值
    attackBonus := m.calculateAttackBonus(attacker, weapon)

    // 攻击检定
    result.AttackRoll = dice.AbilityCheck(attackBonus, target.ArmorClass, false, false)

    // 检查暴击
    result.Critical = result.AttackRoll.Roll == 20

    // 判断是否命中
    if result.AttackRoll.Roll == 1 {
        // 大失败
        result.Hit = false
        return result, nil
    }

    if result.AttackRoll.Roll == 20 || result.AttackRoll.Total >= target.ArmorClass {
        result.Hit = true
    }

    if !result.Hit {
        return result, nil
    }

    // 计算伤害
    damageDice := weapon.Damage
    if result.Critical {
        // 暴击翻倍骰子
        damageDice = doubleDice(damageDice)
    }

    damageRoller, _ := dice.Parse(damageDice)
    result.DamageRoll = damageRoller.Roll(m.dice.Rand())
    result.DamageType = weapon.DamageType

    // 应用伤害修正
    damageBonus := m.calculateDamageBonus(attacker, weapon)
    result.DamageDealt = result.DamageRoll.Total + damageBonus

    // 应用伤害
    m.applyDamage(target, result.DamageDealt, result.DamageType)

    result.TargetHP = target.CurrentHitPoints
    result.TargetMaxHP = target.MaxHitPoints
    result.TargetKilled = target.CurrentHitPoints <= 0

    // 触发战斗事件
    m.emitCombatEvent("attack", params.AttackerID, params.TargetID, result)

    return result, nil
}

func (m *Manager) calculateAttackBonus(char *models.Character, weapon *models.Weapon) int {
    bonus := char.ProficiencyBonus

    // 武器熟练
    if m.isProficientWithWeapon(char, weapon) {
        bonus += char.ProficiencyBonus
    }

    // 属性修正
    if weapon.IsRanged {
        bonus += dice.GetModifier(char.AbilityScores["dexterity"])
    } else if weapon.IsFinesse {
        // 灵巧武器：力量或敏捷取高
        strMod := dice.GetModifier(char.AbilityScores["strength"])
        dexMod := dice.GetModifier(char.AbilityScores["dexterity"])
        bonus += max(strMod, dexMod)
    } else {
        bonus += dice.GetModifier(char.AbilityScores["strength"])
    }

    return bonus
}

func (m *Manager) applyDamage(target *models.Character, damage int, damageType string) {
    // 检查抗性/免疫
    resistance := m.hasResistance(target, damageType)
    immunity := m.hasImmunity(target, damageType)

    if immunity {
        return // 无伤害
    }

    if resistance {
        damage = damage / 2
    }

    // 先消耗临时 HP
    if target.TemporaryHitPoints > 0 {
        if target.TemporaryHitPoints >= damage {
            target.TemporaryHitPoints -= damage
            return
        }
        damage -= target.TemporaryHitPoints
        target.TemporaryHitPoints = 0
    }

    // 消耗正常 HP
    target.CurrentHitPoints -= damage

    // 检查死亡
    if target.CurrentHitPoints <= 0 {
        target.CurrentHitPoints = 0
        target.Conditions = append(target.Conditions, models.ConditionUnconscious)
        target.DeathSaves = &models.DeathSaves{}
    }
}
```
